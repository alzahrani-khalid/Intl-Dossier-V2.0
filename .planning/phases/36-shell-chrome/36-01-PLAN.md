    import userEvent from '@testing-library/user-event'
    import { describe, it, expect } from 'vitest'
    import { Drawer } from '@heroui/react'
    import { useState } from 'react'

    function TwoDrawers(): JSX.Element {
      const [outerOpen, setOuterOpen] = useState(true)
      const [innerOpen, setInnerOpen] = useState(true)
      return (
        <>
          <Drawer isOpen={outerOpen} onOpenChange={setOuterOpen} placement="right">
            <input aria-label="outer-input" />
            <button onClick={() => setInnerOpen(true)}>open-inner</button>
          </Drawer>
          <Drawer isOpen={innerOpen} onOpenChange={setInnerOpen} placement="left">
            <input aria-label="inner-input" />
            <button aria-label="inner-close" onClick={() => setInnerOpen(false)}>x</button>
          </Drawer>
        </>
      )
    }

    describe('Concurrent HeroUI Drawers (Pitfall 3 — A1 validation)', () => {
      it('keeps focus contained inside the most-recently opened drawer', async () => {
        const user = userEvent.setup()
        render(<TwoDrawers />)
        const innerInput = await screen.findByLabelText('inner-input')
        innerInput.focus()
        expect(document.activeElement).toBe(innerInput)
        await user.tab()
        await user.tab()
        await user.tab()
        const stillInside =
          innerInput.closest('[role="dialog"],[data-slot="content"]')?.contains(
            document.activeElement as Node,
          ) ?? false
        expect(stillInside).toBe(true)

      it('ESC closes the innermost drawer first (React Aria restoration)', async () => {

        render(<TwoDrawers />)
        const innerInput = await screen.findByLabelText('inner-input')
        innerInput.focus()
        await user.keyboard('{Escape}')
        expect(screen.queryByLabelText('inner-input')).toBeNull()
        expect(screen.queryByLabelText('outer-input')).not.toBeNull()
      })
