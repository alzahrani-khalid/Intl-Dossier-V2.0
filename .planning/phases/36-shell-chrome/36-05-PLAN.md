    ```ts
    import { test, expect } from '@playwright/test'

    test.describe('Phase 36 shell', () => {
      test.beforeEach(async ({ page }) => {
        // Use test creds from env (project convention)
        await page.goto('/login')
        await page.fill('[name="email"]', process.env.TEST_USER_EMAIL ?? '')
        await page.fill('[name="password"]', process.env.TEST_USER_PASSWORD ?? '')
        await page.click('button[type="submit"]')
        await page.waitForURL('**/')
      })

      test('shell no remount — route changes do not unmount AppShell', async ({ page }) => {
        // Mark the shell mount node so we can observe it survives across navigations
        await page.evaluate(() => {
          const shell = document.querySelector('.appshell')
          if (shell) shell.setAttribute('data-test-mount-id', String(Math.random()))
        })
        const before = await page.getAttribute('.appshell', 'data-test-mount-id')
        // Navigate to another protected route
        await page.goto('/engagements')
        const after = await page.getAttribute('.appshell', 'data-test-mount-id')
        expect(after).toBe(before) // Same instance — did not remount
      })

      test('direction atomic — switching direction does not flicker', async ({ page }) => {
        // Baseline: chancery
        const initialColor = await page.evaluate(() =>
          getComputedStyle(document.documentElement).getPropertyValue('--accent'),
        )
        // Switch to situation via topbar button
        await page.click('.tb-dir-btn:has-text("Situation")')
        // Wait one frame
        await page.waitForFunction(() => {
          const c = getComputedStyle(document.documentElement).getPropertyValue('--accent')
          return c && c.trim() !== ''
        })
        const afterColor = await page.evaluate(() =>
          getComputedStyle(document.documentElement).getPropertyValue('--accent'),
        )
        expect(afterColor.trim()).not.toBe(initialColor.trim())
        // Shell still mounted
        const shell = page.locator('.appshell')
        await expect(shell).toBeVisible()
      })

      test('shell tab order — tab cycles through topbar controls', async ({ page }) => {
        await page.goto('/')
        // Focus the body, then tab — expect first focusable is in topbar
        await page.keyboard.press('Tab')
        const first = await page.evaluate(() => document.activeElement?.className ?? '')
        // Either hamburger (mobile) or search input (desktop)
        expect(first).toMatch(/tb-menu|tb-search|tb-search-input/)
      })
    })
    ```
