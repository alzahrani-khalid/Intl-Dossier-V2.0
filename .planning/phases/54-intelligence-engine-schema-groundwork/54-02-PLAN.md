The RLS pattern (Pattern 1, 54-PATTERNS.md L132-175) for each table:

```sql
ALTER TABLE public.<table> ENABLE ROW LEVEL SECURITY;

DROP POLICY IF EXISTS <table>_select_org ON public.<table>;
CREATE POLICY <table>_select_org
  ON public.<table> FOR SELECT
  TO authenticated
  USING (tenant_isolation.rls_select_policy(organization_id));

DROP POLICY IF EXISTS <table>_insert_editor ON public.<table>;
CREATE POLICY <table>_insert_editor
  ON public.<table> FOR INSERT
  TO authenticated
  WITH CHECK (
    tenant_isolation.rls_insert_policy(organization_id)
    AND public.auth_has_any_role(ARRAY['admin', 'editor'])
  );

DROP POLICY IF EXISTS <table>_update_editor ON public.<table>;
CREATE POLICY <table>_update_editor
  ON public.<table> FOR UPDATE
  TO authenticated
  USING (
    tenant_isolation.rls_update_policy(organization_id)
    AND public.auth_has_any_role(ARRAY['admin', 'editor'])
  )
  WITH CHECK (
    tenant_isolation.rls_update_policy(organization_id)
    AND public.auth_has_any_role(ARRAY['admin', 'editor'])
  );

DROP POLICY IF EXISTS <table>_delete_admin ON public.<table>;
CREATE POLICY <table>_delete_admin
  ON public.<table> FOR DELETE
  TO authenticated
  USING (tenant_isolation.rls_delete_policy(organization_id));  -- helper-only; no extra AND

GRANT SELECT ON public.<table> TO authenticated;
GRANT INSERT, UPDATE, DELETE ON public.<table> TO authenticated;
```
