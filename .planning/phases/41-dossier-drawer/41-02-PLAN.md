    ```typescript
    import { useTranslation } from 'react-i18next'
    import { differenceInCalendarDays } from 'date-fns'
    import { Icon } from '@/components/signature-visuals'
    import { toArDigits } from '@/lib/i18n/toArDigits'
    import { LtrIsolate } from '@/components/ui/ltr-isolate'

    export interface DrawerMetaStripProps {
      dossierId: string
      metadata?: { region?: string; lead_name?: string; [k: string]: unknown } | undefined
      updatedAt?: string | undefined
      engagementCount: number
    }

    const DOT = '·' // middle-dot separator

    export function DrawerMetaStrip(props: DrawerMetaStripProps): JSX.Element {
      const { metadata, updatedAt, engagementCount } = props
      const { t, i18n } = useTranslation('dossier-drawer')
      const lang = i18n.language

      const location = metadata?.region ?? t('meta.location_fallback')
      const leadName = typeof metadata?.lead_name === 'string' && metadata.lead_name.length > 0 ? metadata.lead_name : null
      const leadSegment = leadName !== null
        ? `${t('meta.lead_prefix')}: ${leadName}`
        : '—'

      const engagementSegment = `${toArDigits(engagementCount, lang)} ${t('meta.engagements_suffix')}`

      let lastTouched: string
      if (typeof updatedAt === 'string' && updatedAt.length > 0) {
        const days = differenceInCalendarDays(new Date(), new Date(updatedAt))
        if (days <= 0) lastTouched = t('meta.last_touched_today')
        else if (days === 1) lastTouched = t('meta.last_touched_yesterday')
        else lastTouched = t('meta.last_touched_relative', { n: toArDigits(days, lang) })
      } else {
        lastTouched = t('meta.last_touched_today')
      }

      return (
        <div className="drawer-meta flex items-center gap-2 flex-wrap">
          <span className="inline-flex items-center gap-1">
            <Icon name="map-pin" size={14} />
            {location}
          </span>
          <span aria-hidden="true">{DOT}</span>
          <span>{leadSegment}</span>
          <span aria-hidden="true">{DOT}</span>
          <LtrIsolate>{engagementSegment}</LtrIsolate>
          <span aria-hidden="true">{DOT}</span>
          <span>{lastTouched}</span>
        </div>
      )
    }
    ```

    **Decision references:** RESEARCH §13 (location source = `metadata.region`, lead source = `metadata.lead_name`, last touched uses `formatRelativeTimeShort` semantics but inline differenceInCalendarDays here for the labeled variants); UI-SPEC "Color" anti-list (no emoji — replaces handoff `📍` with `<Icon name="map-pin" />`); `LtrIsolate` keeps Arabic-Indic digits LTR inside the RTL line.
