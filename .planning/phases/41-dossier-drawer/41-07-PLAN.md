    test('AR locale — drawer renders at physical LEFT edge (inline-end maps to left under RTL)', async ({ page }) => {
      await loginForListPages(page)
      await page.evaluate(() => {
        localStorage.setItem('id.locale', 'ar')
        localStorage.setItem('i18nextLng', 'ar')
      })
      await openDrawerForFixtureDossier(page, { id: FIXTURE_DOSSIER_ID, type: 'country' })
      const styles = await page.locator('.drawer').evaluate((el) => ({
        insetInlineEnd: getComputedStyle(el).insetInlineEnd,
        rectLeft: el.getBoundingClientRect().left,
        rectRight: el.getBoundingClientRect().right,
        viewportWidth: window.innerWidth,
        dir: getComputedStyle(el).direction,
      }))
      expect(styles.dir).toBe('rtl')
      expect(styles.insetInlineEnd).toBe('0px')
      expect(styles.rectLeft).toBe(0)  // RTL: inline-end resolves to physical LEFT edge

    test('EN locale — drawer renders at physical RIGHT edge (sanity for inverse)', async ({ page }) => {
      await loginForListPages(page)
