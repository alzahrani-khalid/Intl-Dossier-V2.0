    import { useTranslation } from 'react-i18next'
    import type { DossierOverviewResponse } from '@/types/dossier-overview.types'
    import { toArDigits } from '@/lib/i18n/toArDigits'
    import { LtrIsolate } from '@/components/ui/ltr-isolate'

    export interface MiniKpiStripProps {
      overview?: DossierOverviewResponse | undefined
    }

    /**
     * Verbatim port of handoff app.css#L298-302 + pages.jsx kpi-mini-strip.
     * D-04 KPI mapping (LOCKED):
     *   engagements → stats.calendar_events_count
     *   commitments → work_items.by_source.commitments.length
     *   overdue     → stats.overdue_work_items
     *   documents   → stats.documents_count
     */
    export function MiniKpiStrip({ overview }: MiniKpiStripProps): JSX.Element {
      const { t, i18n } = useTranslation('dossier-drawer')
      const lang = i18n.language

      const engagements = overview?.stats.calendar_events_count ?? 0
      const commitments = overview?.work_items.by_source.commitments?.length ?? 0
      const overdue = overview?.stats.overdue_work_items ?? 0
      const documents = overview?.stats.documents_count ?? 0

      const cells: ReadonlyArray<{ key: 'engagements' | 'commitments' | 'overdue' | 'documents'; val: number }> = [
        { key: 'engagements', val: engagements },
        { key: 'commitments', val: commitments },
        { key: 'overdue', val: overdue },
        { key: 'documents', val: documents },
      ]

      return (
        <div className="kpi-mini-strip" data-testid="dossier-drawer-kpi-strip">
          {cells.map((c) => (
            <div key={c.key} className="kpi-mini">
              <span className="kpi-mini-val">
                <LtrIsolate>{toArDigits(c.val, lang)}</LtrIsolate>
              </span>
              <span className="kpi-mini-label">{t(`kpi.${c.key}`)}</span>
            </div>
          ))}
        </div>
      )
    }
    ```

    **Decision references:** D-04 (KPI mapping locked verbatim), UI-SPEC Typography (kpi-mini-val 22px display, kpi-mini-label 11px ink-mute), RESEARCH §2 (field names verified). NOTE on D-04 fallback: per CONTEXT.md, if `calendar_events_count` proves not to track engagement-type events accurately, planner authorizes a fallback filter on `work_items` by `engagement_type`. This fallback is captured as deferred — Phase 41 ships with the locked mapping; manual verification (UI-SPEC Manual-Only Verifications row "Calendar prefill mechanism (A1)") flags any real-data drift in Wave 2.

    import { useTranslation } from 'react-i18next'

    export interface DossierForSummary {
      description_en?: string | null
      description_ar?: string | null
    }

    export interface SummarySectionProps {
      dossier?: DossierForSummary | undefined
    }

    /**
     * Verbatim port of handoff app.css#L516.
     * Section label uppercase per CLAUDE.md voice rule (UPPERCASE only for labels/ribbons/columns).
     * Summary paragraph: italic display font, ink-mute, 14px, line-height 1.6.
     */
    export function SummarySection({ dossier }: SummarySectionProps): JSX.Element {
      const { t, i18n } = useTranslation('dossier-drawer')
      const lang = i18n.language

      const enText = dossier?.description_en ?? ''
      const arText = dossier?.description_ar ?? ''
      let body: string
      if (lang === 'ar' && arText.length > 0) body = arText
      else if (enText.length > 0) body = enText
      else if (arText.length > 0) body = arText // fallback when EN missing
      else body = t('empty.summary')

      return (
        <section className="flex flex-col gap-2" data-testid="dossier-drawer-summary">
          <h3
            className="t-label"
            style={{
              fontSize: 'var(--t-label, 10.5px)',
              fontWeight: 600,
              letterSpacing: 'var(--tracking-label, 0.1em)',
              color: 'var(--ink-mute)',
              textTransform: 'uppercase',
            }}
          >
            {t('section.summary')}
          </h3>
          <p
            style={{
              fontFamily: 'var(--font-display)',
              fontStyle: 'italic',
              color: 'var(--ink-mute)',
              fontSize: '14px',
              lineHeight: 1.6,
            }}
          >
            {body}
          </p>
        </section>
      )
    }
    ```

    **Decision references:** UI-SPEC Typography (Summary paragraph token row), Copywriting Contract (section.summary + empty.summary keys), CLAUDE.md (italic only on Summary; UPPERCASE only for labels). The fallback chain (AR locale prefers AR text → falls back to EN → falls back to empty.summary key) handles the realistic data shape where many dossiers will only have EN descriptions seeded.

    No `text-left/right` — paragraph relies on parent `dir` for alignment. No raw hex.
