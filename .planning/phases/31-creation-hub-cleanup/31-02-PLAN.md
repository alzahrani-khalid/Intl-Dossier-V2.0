    ```tsx
    import { useState } from 'react'
    import { useTranslation } from 'react-i18next'
    import { Info } from 'lucide-react'
    // Try HeroUI v3 first per CLAUDE.md hierarchy:
    import { Alert } from '@heroui/react'
    // If @heroui/react exports no Alert (or API differs materially), fall back to
    // `import { Alert, AlertDescription } from '@/components/ui/alert'` — the shadcn
    // wrapper already used in DossierListPage. Document the chosen path in the SUMMARY.

    interface StepGuidanceBannerProps {
      type: string       // DossierType as string (e.g. 'country', 'elected_official')
      stepId: string     // e.g. 'basic', 'country-details', 'review'
      guidanceKey: string // full i18n key including namespace, e.g. 'country-wizard:wizard.steps.basic.guidance'
    }

    export function StepGuidanceBanner({ type, stepId, guidanceKey }: StepGuidanceBannerProps): JSX.Element | null {
      const { t } = useTranslation()
      const storageKey = `dossier-wizard:guidance:${type}:${stepId}`
      const [dismissed, setDismissed] = useState<boolean>(() => {
        try { return localStorage.getItem(storageKey) === '1' } catch { return false }
      })
      if (dismissed) return null

      const handleDismiss = (): void => {
        try { localStorage.setItem(storageKey, '1') } catch { /* ignore */ }
        setDismissed(true)
      }

      return (
        <Alert
          variant="flat"
          color="primary"
          startContent={<Info className="h-4 w-4" />}
          onClose={handleDismiss}
          isClosable
          className="mb-4 text-start"
          data-testid={`guidance-banner-${type}-${stepId}`}
        >
          {t(guidanceKey)}
        </Alert>
      )
    }
    ```

    **RTL safety:** `className="mb-4 text-start"` — no `ml-*`, `text-left`. The `<Info />` icon is non-directional so no `rotate-180` logic needed (per CLAUDE.md RTL discretion).

    **Security:** render translated text via `{t(guidanceKey)}` only — React escapes by default. Do NOT use any raw-HTML injection prop.
