### Task 1: add false-positive filter to assertTouchTargets

Edit `frontend/tests/e2e/qa-sweep-responsive.spec.ts`. Inside the
`page.evaluate` body of `assertTouchTargets`, add three predicates BEFORE
the rect-size check:

```ts
const isAriaHidden = (el: Element): boolean => {
  let cur: Element | null = el
  while (cur && cur !== document.body) {
    if ((cur as HTMLElement).getAttribute('aria-hidden') === 'true') return true
    cur = cur.parentElement
  }
  return false
}
const isSrOnly = (el: Element): boolean => {
  const cs = getComputedStyle(el as Element)
  // Tailwind .sr-only: width 1px, height 1px, position absolute, overflow hidden, clip-path inset(50%)
  const w = parseFloat(cs.width)
  const h = parseFloat(cs.height)
  const tinyClipped =
    (w <= 1 || h <= 1) && (cs.overflow === 'hidden' || cs.clipPath.includes('inset'))
  if (tinyClipped) return true
  // Legacy clip rect(0,0,0,0)
  if (cs.clip === 'rect(0px, 0px, 0px, 0px)' || cs.clip === 'rect(0, 0, 0, 0)') return true
  return false
}
const hasPointerEventsNone = (el: Element): boolean => {
  let cur: Element | null = el
  while (cur && cur !== document.body) {
    if (getComputedStyle(cur as Element).pointerEvents === 'none') return true
    cur = cur.parentElement
  }
  return false
}
```

Then update the `.filter()` chain to short-circuit these BEFORE the rect
predicate:

```ts
.filter((el) => {
  if (isAriaHidden(el)) return false
  if (isSrOnly(el)) return false
  if (hasPointerEventsNone(el)) return false
  const cs = getComputedStyle(el as Element)
  if (cs.display === 'none' || cs.visibility === 'hidden') return false
  const rect = (el as HTMLElement).getBoundingClientRect()
  return rect.width > 0 && rect.height > 0 && (rect.width < min || rect.height < min)
})
```
