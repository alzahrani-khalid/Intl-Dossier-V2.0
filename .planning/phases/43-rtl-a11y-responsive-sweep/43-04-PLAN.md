```ts
import { test, expect } from '@playwright/test'
import { loginForListPages } from './support/list-pages-auth'
import { settlePage, assertFocusOutlineVisible } from './helpers/qa-sweep'

const DIRECTIONS = ['chancery', 'situation', 'ministerial', 'bureau'] as const
const MODES = ['light', 'dark'] as const

// Representative primitive on Settings — choose first <button> in main settings nav
// (Phase 33-07 default tab is 'general', so settings-nav buttons are stable on initial paint).
const PRIMITIVE_SELECTOR = 'main button:visible, main a[href]:visible, main input:visible'

test.describe('Phase 43 — qa-sweep-focus-outline (Settings × 4 dirs × 2 modes)', () => {
  for (const direction of DIRECTIONS) {
    for (const mode of MODES) {
      test(`settings [${direction}/${mode}] — focus outline visible + ≥3:1 contrast`, async ({
        page,
      }) => {
        await loginForListPages(page, 'en')
        await page.goto('/settings')
        await settlePage(page)

        // Wait for the test hatch (env-gated to DEV/test).
        await page.waitForFunction(
          () =>
            typeof (window as unknown as { __design?: { setDirection?: unknown } }).__design
              ?.setDirection === 'function',
          { timeout: 5000 },
        )

        // Apply direction + mode via hatch.
        await page.evaluate(
          ([d, m]) => {
            const hatch = (
              window as unknown as {
                __design: { setDirection: (d: string) => void; setMode: (m: string) => void }
              }
            ).__design
            hatch.setDirection(d)
            hatch.setMode(m)
          },
          [direction, mode] as const,
        )

        // Settle paint after token swap.
        await page.waitForTimeout(150)

        // Focus the first visible primitive in <main>.
        const target = page.locator(PRIMITIVE_SELECTOR).first()
        await target.focus()
        await page.waitForTimeout(50)

        // Programmatic contrast assertion (returns ratio + colors for snapshot context).
        const probe = await assertFocusOutlineVisible(page, PRIMITIVE_SELECTOR)
        // Visual baseline.
        await expect(page).toHaveScreenshot(`${direction}-${mode}-focused-primitive.png`, {
          // Match playwright.config defaults but pin a focused crop for review.
          maxDiffPixelRatio: 0.01,
          animations: 'disabled',
          caret: 'hide',
        })

        // Surface diagnostics in CI logs.
        console.log(
          `[settings][${direction}/${mode}] outline=${probe.outlineColor} bg=${probe.bgColor} ratio=${probe.ratio.toFixed(2)}`,
        )
      })
    }
  }

  test.afterEach(async ({ page }) => {
    // Reset to v6.0 defaults so subsequent tests aren't poisoned (per RESEARCH §5 pitfall).
    await page
      .evaluate(() => {
        const hatch = (
          window as unknown as {
            __design?: {
              setDirection: (d: string) => void
              setMode: (m: string) => void
              setHue: (h: number) => void
              setDensity: (d: string) => void
            }
          }
        ).__design
        if (hatch) {
          hatch.setDirection('bureau')
          hatch.setMode('light')
          hatch.setHue(32)
          hatch.setDensity('comfortable')
        }
      })
      .catch(() => {})
  })
})
```

Notes:

- Locale stays `en` for all 8 baselines per D-08 (locale is NOT part of the matrix).
- `PRIMITIVE_SELECTOR` is intentionally generic — `assertFocusOutlineVisible` returns the focused element it finds. If a particular direction/mode renders no `<main button>`, the test fails fast on the focus call (real bug to fix in Plan 43-07).
- Baselines are NOT pre-committed by this plan. The first run will fail with "snapshot not found" — Plan 43-07 runs `--update-snapshots`, then human reviews and commits 8 PNGs.
- `test.afterEach` resets the hatch to v6.0 defaults to avoid poisoning subsequent tests (RESEARCH §5 pitfall).
