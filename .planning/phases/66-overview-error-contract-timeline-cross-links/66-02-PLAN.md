```typescript
// frontend/src/lib/timeline-navigation.ts
// OVRERR-02: server/DB-sourced navigation URLs must resolve to MOUNTED routes.
// Folds in the R-05/WR-01 URL-safety rules from ActivityList (relative-only,
// // and \ rejection). Static allowlist — see RESEARCH A1 for why not matchRoute.
const MOUNTED_DETAIL_PREFIXES: readonly RegExp[] = [
  /^\/tasks\/[^/]+$/,
  /^\/intake\/tickets\/[^/]+$/,
  /^\/positions\/[^/]+/,
  /^\/engagements\/[^/]+/,
  /^\/after-actions\/[^/]+/,
  /^\/dossiers\/(countries|organizations|forums|topics|working_groups|persons|elected-officials|engagements)\/[^/]+/,
  /^\/commitments(\?|$)/,
  /^\/mous$/,
  /^\/calendar$/,
  /^\/activity$/,
]

export function resolveTimelineNavUrl(raw: unknown): string | null {
  if (typeof raw !== 'string' || raw === '') return null
  if (!raw.startsWith('/') || raw.startsWith('//') || raw.includes('\\')) return null
  const pathOnly = raw.split('?')[0] ?? raw
  return MOUNTED_DETAIL_PREFIXES.some((re) => re.test(pathOnly) || re.test(raw)) ? raw : null
}
```

```typescript
function getWorkItemUrl(item: QuickSwitcherWorkItem): string | null {
  switch (item.type) {
    case 'position':
      return `/positions/${item.id}`
    case 'task':
      return `/tasks/${item.id}`
    case 'commitment':
      return `/commitments?id=${item.id}`
    case 'intake':
      return `/intake/tickets/${item.id}`
    case 'mou':
      // /mous/$id is UNMOUNTED. Org dossiers have a mounted mous tab; otherwise the list.
      return item.dossier_context?.type === 'organization'
        ? `${getDossierDetailPath(item.dossier_context.id, 'organization')}/mous`
        : '/mous'
    case 'document':
      // No /documents/* route exists. Mounted docs tabs exist for the 7 dossier types
      // (/dossiers/{segment}/$id/docs) but NOT under /dossiers/engagements — suppress those.
      return item.dossier_context != null && item.dossier_context.type !== 'engagement'
        ? `${getDossierDetailPath(item.dossier_context.id, item.dossier_context.type)}/docs`
        : null
    default:
      return null // was '#' — a dead affordance; null lets callers suppress
  }
}
```
