  <name>Task 2: Write populate_diplomatic_seed migration</name>
  <files>supabase/migrations/20260407000002_populate_diplomatic_seed.sql</files>
  <action>
    Create migration with this structure:

    ```sql
    -- Phase 17: GASTAT diplomatic seed (D-06, D-07, D-13, D-14)
    -- Canonical tables: tasks (NOT work_items), persons (no separate elected_officials).
    -- Bilingual: native *_ar/*_en where available; tasks use title=EN, description=AR+context (§3 Option A).
    -- Idempotent on empty only. Admin-only. Deterministic UUIDs.

    CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

    CREATE OR REPLACE FUNCTION public.populate_diplomatic_seed()
    RETURNS jsonb
    LANGUAGE plpgsql
    SECURITY DEFINER
    SET search_path = public
    AS $$
    DECLARE
      v_caller UUID := auth.uid();
      v_tenant_id UUID;
      v_is_admin BOOLEAN;
      v_already BOOLEAN;
      v_seed_ns CONSTANT UUID := 'a0eebc99-9c0b-4ef8-bb6d-6bb9bd380a17'; -- seed.gastat.diplomatic
      v_counts jsonb;
    BEGIN
      -- 1. Auth gate
      IF v_caller IS NULL THEN
        RETURN jsonb_build_object('status', 'forbidden', 'reason', 'unauthenticated');
      END IF;

      -- 2. Admin-only role gate (strict admin per D-11)
      SELECT EXISTS (
        SELECT 1 FROM user_roles
        WHERE user_id = v_caller AND role = 'admin'
      ) INTO v_is_admin;

      IF NOT v_is_admin THEN
        RETURN jsonb_build_object('status', 'forbidden', 'reason', 'not_admin');
      END IF;

      -- 3. Tenant resolution (§6) — fail loudly rather than insert orphans
      SELECT tenant_id INTO v_tenant_id
      FROM public.users
      WHERE id = v_caller;

      IF v_tenant_id IS NULL THEN
        RAISE EXCEPTION 'populate_diplomatic_seed: no tenant for caller %', v_caller
          USING HINT = 'Assign the admin user to a tenant before running seed.';
      END IF;

      -- 4. Idempotency gate (D-13)
      SELECT EXISTS (
        SELECT 1 FROM countries WHERE is_seed_data = true
        UNION ALL SELECT 1 FROM organizations WHERE is_seed_data = true
        UNION ALL SELECT 1 FROM dossiers WHERE is_seed_data = true
      ) INTO v_already;

      IF v_already THEN
        RETURN jsonb_build_object('status', 'already_seeded');
      END IF;

      -- 5. INSERTS with deterministic UUIDs via uuid_generate_v5(v_seed_ns, '<key>')
      --    Sectioned: countries → orgs → forums → topics → working_groups →
      --               persons (incl. person_subtype='elected_official') →
      --               dossiers (all 8 types) → engagements → tasks → work_item_dossiers
      --    Every row sets is_seed_data = true.
      --    tasks: title=EN, description=AR+context; tenant_id=v_tenant_id;
      --           assignee_id/created_by/last_modified_by=v_caller;
      --           workflow_stage explicit; source jsonb varies across
      --           {kind:'task'}, {kind:'commitment',commitment_id:...}, {kind:'intake',ticket_id:...};
      --           work_item_type='task', work_item_id=NULL.
      --    work_item_dossiers: work_item_type='task', varied inheritance_source,
      --           exactly one is_primary=true per task, sequential display_order per dossier.

      -- ... full fixture INSERTs here ...

      -- 6. Counts
      v_counts := jsonb_build_object(
        'countries', (SELECT count(*) FROM countries WHERE is_seed_data = true),
        'organizations', (SELECT count(*) FROM organizations WHERE is_seed_data = true),
        'forums', (SELECT count(*) FROM forums WHERE is_seed_data = true),
        'engagements', (SELECT count(*) FROM engagements WHERE is_seed_data = true),
        'dossiers', (SELECT count(*) FROM dossiers WHERE is_seed_data = true),
        'tasks', (SELECT count(*) FROM tasks WHERE is_seed_data = true),
        'topics', (SELECT count(*) FROM topics WHERE is_seed_data = true),
        'working_groups', (SELECT count(*) FROM working_groups WHERE is_seed_data = true),
        'persons', (SELECT count(*) FROM persons WHERE is_seed_data = true)
      );

      RETURN jsonb_build_object('status', 'seeded', 'counts', v_counts);
    END;
    $$;

    GRANT EXECUTE ON FUNCTION public.populate_diplomatic_seed() TO authenticated;

    COMMENT ON FUNCTION public.populate_diplomatic_seed() IS
      'Phase 17 GASTAT diplomatic seed. Admin-only, idempotent on empty, deterministic UUIDs. Uses canonical tasks table (not work_items) and persons.person_subtype for elected officials.';
    ```

    Fill in the INSERT sections from the fixture design in Task 1. Use `uuid_generate_v5(v_seed_ns, 'country.sa')` style keys so every row has a stable UUID. Always set `is_seed_data = true`. Use `ON CONFLICT (id) DO NOTHING` as a defensive guard.
