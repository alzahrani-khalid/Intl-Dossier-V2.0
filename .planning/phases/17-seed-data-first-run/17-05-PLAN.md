  <action>
    Create `useFirstRunCheck.ts`:

    ```ts
    import { useQuery } from '@tanstack/react-query'
    import { supabase } from '@/lib/supabase' // verify exact import path

    export interface FirstRunStatus {
      isEmpty: boolean
      canSeed: boolean
    }

    export const useFirstRunCheck = (): {
      data: FirstRunStatus | undefined
      isLoading: boolean
      isError: boolean
    } => {
      const query = useQuery({
        queryKey: ['first-run-check'],
        queryFn: async (): Promise<FirstRunStatus> => {
          const { data, error } = await supabase.rpc('check_first_run')
          if (error) throw error
          const payload = data as { is_empty: boolean; can_seed: boolean }
          return { isEmpty: payload.is_empty, canSeed: payload.can_seed }
        },
        staleTime: Infinity, // never refetch automatically — single fetch per session
        gcTime: Infinity,
        retry: 1,
      })
      return { data: query.data, isLoading: query.isLoading, isError: query.isError }
    }
    ```

    Add a Vitest test mocking `supabase.rpc` and asserting the hook maps `is_empty`/`can_seed` → `isEmpty`/`canSeed`, and that `staleTime: Infinity` is applied (verify by re-rendering and confirming no refetch).

  <files>frontend/src/routes/_protected/dashboard.tsx</files>
  <action>
    Read the file first to identify the dashboard component name and a stable insertion point inside the JSX (after auth gate, alongside other top-level widgets).

    Add at the top of the component body:

    ```tsx
    const { data: firstRun } = useFirstRunCheck()
    const [dismissed, setDismissed] = useState<boolean>(() => {
      if (typeof window === 'undefined') return false
      return window.localStorage.getItem('intl-dossier:first-run-dismissed') === 'true'
    })

    const showFirstRun = firstRun?.isEmpty === true && !dismissed

    const handleOpenChange = (open: boolean): void => {
      if (!open) {
        setDismissed(true)
        window.localStorage.setItem('intl-dossier:first-run-dismissed', 'true')
      }
    }
    ```

    Then render at the end of the dashboard JSX (sibling of other widgets):

    ```tsx
    <FirstRunModal
      open={showFirstRun}
      onOpenChange={handleOpenChange}
      canSeed={firstRun?.canSeed ?? false}
    />
    ```

    Add the imports for `useState`, `useFirstRunCheck`, and `FirstRunModal`. Do not change any other behavior in the dashboard route.

    **localStorage key contract:** `intl-dossier:first-run-dismissed` (string `'true'`).

    **Re-show after successful seed:** when the populate mutation succeeds, the FirstRunModal closes itself via `onOpenChange(false)`, which sets the dismissed flag — that is correct: after seeding, the DB is no longer empty, so even if the user clears localStorage, `firstRun.isEmpty` will be false on next load.
