```tsx
/**
 * Countries List Page (Phase 40 LIST-01)
 *
 * Replaces legacy implementation per .planning/phases/40-list-pages/40-CONTEXT.md D-01.
 * Renders DossierTable inside ListPageShell with useCountries adapter.
 */

import { createFileRoute } from '@tanstack/react-router'
import { useTranslation } from 'react-i18next'
import { useCallback, useMemo } from 'react'
import {
  ListPageShell,
  DossierTable,
  DossierTableSkeleton,
  ToolbarSearch,
} from '@/components/list-page'
import type { DossierTableRow } from '@/components/list-page'
import { useCountries } from '@/hooks/useCountries'
import { useDirection } from '@/hooks/useDirection'

interface DossierListSearch {
  page: number
  search?: string
}

export const Route = createFileRoute('/_protected/dossiers/countries/')({
  component: CountriesListPage,
  validateSearch: (search: Record<string, unknown>): DossierListSearch => ({
    page: Math.max(1, Number(search.page) || 1),
    search:
      typeof search.search === 'string' && search.search.length > 0 ? search.search : undefined,
  }),
})

function CountriesListPage(): JSX.Element {
  const { t } = useTranslation(['countries', 'list-pages'])
  const { isRTL } = useDirection()
  const { page, search } = Route.useSearch()
  const navigate = Route.useNavigate()

  const query = useCountries({ page, limit: 20, search })

  const rows: DossierTableRow[] = useMemo(() => {
    const dossiers =
      (query.data as { data?: Array<Record<string, unknown>> } | undefined)?.data ?? []
    return dossiers.map((d) => ({
      id: String(d.id),
      type: 'country' as const,
      name_en: String(d.name_en ?? ''),
      name_ar: String(d.name_ar ?? ''),
      flag_iso: typeof d.iso_code === 'string' ? d.iso_code : undefined,
      engagement_count: typeof d.engagement_count === 'number' ? d.engagement_count : 0,
      updated_at: String(d.updated_at ?? ''),
      sensitivity_level: (typeof d.sensitivity_level === 'number' &&
      d.sensitivity_level >= 1 &&
      d.sensitivity_level <= 4
        ? d.sensitivity_level
        : 1) as 1 | 2 | 3 | 4,
    }))
  }, [query.data])

  const onSearchChange = useCallback(
    (next: string): void => {
      void navigate({
        search: (prev) => ({ ...prev, search: next.length > 0 ? next : undefined, page: 1 }),
      })
    },
    [navigate],
  )

  return (
    <ListPageShell
      title={t('countries:title')}
      subtitle={t('countries:subtitle')}
      toolbar={
        <ToolbarSearch
          value={search ?? ''}
          onChange={onSearchChange}
          placeholder={t('list-pages:search.placeholder')}
        />
      }
      isLoading={query.isLoading}
      isEmpty={!query.isLoading && rows.length === 0}
      emptyState={
        <div className="empty-hint">
          <p>{t('countries:empty.title')}</p>
          <p className="sub">{t('countries:empty.description')}</p>
        </div>
      }
      skeleton={<DossierTableSkeleton rows={8} />}
    >
      <DossierTable rows={rows} isRTL={isRTL} t={t} />
    </ListPageShell>
  )
}
```
