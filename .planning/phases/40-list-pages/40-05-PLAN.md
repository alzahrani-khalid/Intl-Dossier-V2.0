```tsx
/**
 * Persons List Page (Phase 40 LIST-02)
 *
 * Replaces legacy implementation per .planning/phases/40-list-pages/40-CONTEXT.md D-04.
 */

import { useMemo, useState } from 'react'
import { useTranslation } from 'react-i18next'
import {
  ListPageShell,
  PersonsGrid,
  PersonsGridSkeleton,
  ToolbarSearch,
} from '@/components/list-page'
import type { PersonCard } from '@/components/list-page'
import { usePersons } from '@/hooks/usePersons'
import { useDirection } from '@/hooks/useDirection'
import { useDebouncedValue } from '@/hooks/useDebouncedValue'

export default function PersonsListPage(): JSX.Element {
  const { t } = useTranslation(['persons', 'list-pages'])
  const { isRTL } = useDirection()
  const [search, setSearch] = useState('')
  const debouncedSearch = useDebouncedValue(search, 250)
  const query = usePersons({ search: debouncedSearch })

  const items: PersonCard[] = useMemo(() => {
    const raw = query.data as unknown
    let list: Array<Record<string, unknown>> = []
    if (Array.isArray(raw)) {
      list = raw as Array<Record<string, unknown>>
    } else if (
      raw != null &&
      typeof raw === 'object' &&
      'items' in raw &&
      Array.isArray((raw as { items?: unknown[] }).items)
    ) {
      list = (raw as { items: Array<Record<string, unknown>> }).items
    } else if (
      raw != null &&
      typeof raw === 'object' &&
      'data' in raw &&
      Array.isArray((raw as { data?: unknown[] }).data)
    ) {
      list = (raw as { data: Array<Record<string, unknown>> }).data
    }
    return list.map((p) => ({
      id: String(p.id),
      name_en: String(p.name_en ?? p.name ?? ''),
      name_ar: typeof p.name_ar === 'string' ? p.name_ar : undefined,
      role: typeof p.role === 'string' ? p.role : typeof p.title === 'string' ? p.title : undefined,
      organization:
        typeof p.organization_name === 'string'
          ? p.organization_name
          : typeof p.organization === 'string'
            ? p.organization
            : undefined,
      is_vip: p.is_vip === true,
      flag_iso: typeof p.flag_iso === 'string' ? p.flag_iso : undefined,
    }))
  }, [query.data])

  return (
    <ListPageShell
      title={t('persons:title')}
      subtitle={t('persons:subtitle')}
      toolbar={
        <ToolbarSearch
          value={search}
          onChange={setSearch}
          placeholder={t('list-pages:search.placeholder')}
        />
      }
      isLoading={query.isLoading}
      isEmpty={!query.isLoading && items.length === 0}
      emptyState={
        <div className="empty-hint">
          <p>{t('persons:empty.title')}</p>
          <p className="sub">{t('persons:empty.description')}</p>
        </div>
      }
      skeleton={<PersonsGridSkeleton rows={6} />}
    >
      <PersonsGrid items={items} isRTL={isRTL} t={t} />
    </ListPageShell>
  )
}
```

NO `any`. Strict boolean expressions. Logical properties only.
