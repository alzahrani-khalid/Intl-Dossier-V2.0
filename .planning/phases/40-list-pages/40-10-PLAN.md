```ts
import { test, expect } from '@playwright/test'

const ROUTES = [
  { path: '/dossiers/countries', titleKey: 'Countries' },
  { path: '/dossiers/organizations', titleKey: 'Organizations' },
  { path: '/persons', titleKey: 'Persons' },
  { path: '/dossiers/forums', titleKey: 'Forums' },
  { path: '/dossiers/topics', titleKey: 'Topics' },
  { path: '/dossiers/working-groups', titleKey: 'Working Groups' },
  { path: '/engagements', titleKey: 'Engagements' },
] as const

const VIEWPORTS = [
  { width: 320, height: 720, name: 'mobile' },
  { width: 768, height: 1024, name: 'tablet' },
  { width: 1280, height: 800, name: 'desktop' },
] as const

for (const route of ROUTES) {
  for (const vp of VIEWPORTS) {
    test(`${route.path} renders @ ${vp.name}`, async ({ page }) => {
      await page.setViewportSize({ width: vp.width, height: vp.height })
      await page.goto(route.path)
      await expect(page.locator('h1.page-title')).toBeVisible()
      // No horizontal overflow
      const scrollX = await page.evaluate(
        () => document.documentElement.scrollWidth - document.documentElement.clientWidth,
      )
      expect(scrollX).toBeLessThanOrEqual(0)
    })
  }
}

test('Countries table collapses to cards under 768px', async ({ page }) => {
  await page.setViewportSize({ width: 375, height: 720 })
  await page.goto('/dossiers/countries')
  // Desktop table is hidden, mobile cards visible
  await expect(page.locator('.hidden.md\\:block table.tbl').first()).toBeHidden()
  await expect(page.locator('.md\\:hidden').first()).toBeVisible()
})
```

import { test, expect } from '@playwright/test'

test('RTL chevron transform includes scaleX(-1) on /dossiers/countries', async ({ page }) => {
  await page.goto('/dossiers/countries')
  await page.evaluate(() => {
    document.documentElement.dir = 'rtl'
  })
  await page.waitForFunction(() => document.fonts.ready)
  const transform = await page
    .locator('.icon-flip')
    .first()
    .evaluate((el) => getComputedStyle(el).transform)
  // Either matrix(-1, 0, 0, 1, 0, 0) or scaleX(-1) computed form
  expect(transform).toMatch(/matrix\(-1|scaleX\(-1\)/)
})

const ROUTES_WITH_NAMES = [
  ['/dossiers/countries', 'countries'],
  ['/dossiers/organizations', 'organizations'],
  ['/persons', 'persons'],
  ['/dossiers/forums', 'forums'],

        maxDiffPixelRatio: 0.02,
        fullPage: true,
      })
    })
  }
}
