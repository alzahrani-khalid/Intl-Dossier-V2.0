```tsx
/**
 * Engagements List Page (Phase 40 LIST-04)
 *
 * Replaces legacy implementation per .planning/phases/40-list-pages/40-CONTEXT.md D-07/D-08/D-10.
 * Wires the EngagementsList primitive (search + 4 filter pills + week-list + load-more) into ListPageShell.

import { useCallback } from 'react'
import { useNavigate } from '@tanstack/react-router'
import { useTranslation } from 'react-i18next'
import { ListPageShell, EngagementsList } from '@/components/list-page'
import { useDirection } from '@/hooks/useDirection'

export default function EngagementsListPage(): JSX.Element {
  const { t } = useTranslation(['engagements', 'list-pages'])
  const { isRTL } = useDirection()
  const navigate = useNavigate()

  const onRowClick = useCallback(
    (engagementId: string): void => {
      void navigate({ to: '/engagements/$engagementId/overview', params: { engagementId } })
    },
    [navigate],
  )

  return (
    <ListPageShell title={t('engagements:title')} subtitle={t('engagements:subtitle')}>
      <EngagementsList isRTL={isRTL} t={t} onRowClick={onRowClick} />
    </ListPageShell>
  )
}
```

NO `any`. Strict boolean expressions. Logical properties only.
