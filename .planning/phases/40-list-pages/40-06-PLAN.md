```tsx
/**
 * Forums List Page (Phase 40 LIST-03)
 *
 * Replaces legacy implementation per .planning/phases/40-list-pages/40-CONTEXT.md D-01.
 * Renders GenericListPage inside ListPageShell with useForums adapter.
 */

import { createFileRoute } from '@tanstack/react-router'
import { useTranslation } from 'react-i18next'
import { useCallback, useMemo } from 'react'
import {
  ListPageShell,
  GenericListPage,
  GenericListSkeleton,
  ToolbarSearch,
} from '@/components/list-page'
import type { GenericListPageItem } from '@/components/list-page'
import { useForums } from '@/hooks/useForums'
import { useDirection } from '@/hooks/useDirection'

interface DossierListSearch {
  page: number
  search?: string
}

const FORUM_STATUS_TONE: Record<
  string,
  GenericListPageItem['status'] extends infer S ? (S extends { tone: infer T } ? T : never) : never
> = {
  active: 'ok',
  completed: 'info',
  planned: 'accent',
  cancelled: 'danger',
}

export const Route = createFileRoute('/_protected/dossiers/forums/')({
  component: ForumsListPage,
  validateSearch: (search: Record<string, unknown>): DossierListSearch => ({
    page: Math.max(1, Number(search.page) || 1),
    search:
      typeof search.search === 'string' && search.search.length > 0 ? search.search : undefined,
  }),
})

function ForumsListPage(): JSX.Element {
  const { t } = useTranslation(['forums', 'list-pages'])
  const { isRTL } = useDirection()
  const { page, search } = Route.useSearch()
  const navigate = Route.useNavigate()
  const query = useForums({ page, limit: 20, search })

  const items: GenericListPageItem[] = useMemo(() => {
    const list = (query.data as { data?: Array<Record<string, unknown>> } | undefined)?.data ?? []
    return list.map((f) => {
      const status = typeof f.status === 'string' ? f.status : 'active'
      const tone = (FORUM_STATUS_TONE[status] ?? 'info') as
        | 'accent'
        | 'warn'
        | 'danger'
        | 'ok'
        | 'info'
      return {
        id: String(f.id),
        glyphType: 'forum' as const,
        nameEn: String(f.name_en ?? ''),
        nameAr: typeof f.name_ar === 'string' ? f.name_ar : undefined,
        meta:
          typeof f.updated_at === 'string'
            ? new Intl.DateTimeFormat(isRTL ? 'ar-SA' : 'en-US', {
                month: 'short',
                day: 'numeric',
                year: 'numeric',
              }).format(new Date(f.updated_at))
            : undefined,
        status: { label: t(`forums:status.${status}`, { defaultValue: status }), tone },
        href: `/dossiers/forums/${String(f.id)}`,
      }
    })
  }, [query.data, isRTL, t])

  const onSearchChange = useCallback(
    (next: string): void => {
      void navigate({
        search: (prev) => ({ ...prev, search: next.length > 0 ? next : undefined, page: 1 }),
      })
    },
    [navigate],
  )

  return (
    <ListPageShell
      title={t('forums:title')}
      subtitle={t('forums:subtitle')}
      toolbar={
        <ToolbarSearch
          value={search ?? ''}
          onChange={onSearchChange}
          placeholder={t('list-pages:search.placeholder')}
        />
      }
      isLoading={query.isLoading}
      isEmpty={!query.isLoading && items.length === 0}
      emptyState={
        <div className="empty-hint">
          <p>{t('forums:empty.title')}</p>
          <p className="sub">{t('forums:empty.description')}</p>
        </div>
      }
      skeleton={<GenericListSkeleton rows={6} />}
    >
      <GenericListPage items={items} isRTL={isRTL} />
    </ListPageShell>
  )
}
```
