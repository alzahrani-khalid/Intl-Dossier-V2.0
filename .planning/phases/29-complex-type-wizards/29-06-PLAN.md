```ts
import { test, expect } from '@playwright/test'

test.describe('Forum wizard — create flow', () => {
  test('creates a forum via the 3-step wizard and lands on the detail page', async ({ page }) => {
    await page.goto('/dossiers/forums')
    await page.getByRole('link', { name: /create forum/i }).click()
    await expect(page).toHaveURL(/\/dossiers\/forums\/create$/)

    // Basic Info
    await page.getByLabel(/name.*english/i).fill(`E2E Forum ${Date.now()}`)
    await page.getByLabel(/name.*arabic|العربية/i).fill('منتدى اختبار')
    await page.getByRole('button', { name: /next|continue/i }).click()

    // Forum Details — leave organizing body blank (optional per D-10)
    await page.getByRole('button', { name: /next|continue/i }).click()

    // Review + submit
    await page.getByRole('button', { name: /submit|create|finish/i }).click()
    await expect(page).toHaveURL(/\/dossiers\/[0-9a-f-]+$/)
  })
})
```
