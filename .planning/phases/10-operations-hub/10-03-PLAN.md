**1. Create `OperationsHub.tsx`** — Main page component.

Import all zone components, hooks, and types. Set `dir={isRTL ? 'rtl' : 'ltr'}` on root container.

```
export function OperationsHub(): React.ReactElement {
  const { t, i18n } = useTranslation('operations-hub')
  const isRTL = i18n.language === 'ar'
  const { isMobile } = useResponsive()
  const { role, setRole } = useRolePreference()
  const { userId } = useDashboardScope()

  // Zone data hooks — each zone fetches independently (progressive loading)
  const attention = useAttentionItems(userId ?? undefined)
  const events = useGroupedEvents(userId ?? undefined)  // from useUpcomingEvents
  const stages = useEngagementStages(userId ?? undefined)
  const stats = useDashboardStats(userId ?? undefined)
  const activity = useActivityFeed()

  // Zone component map for role-based ordering
  const zoneMap: Record<string, React.ReactElement> = {
    attention: <ZoneCollapsible title={t('zones.attention.title')} defaultExpanded={...} badgeCount={attention.data?.length}>
      <AttentionZone items={attention.data ?? []} isLoading={attention.isLoading} isError={attention.isError} onRetry={() => attention.refetch()} />
    </ZoneCollapsible>,
    timeline: <ZoneCollapsible title={t('zones.timeline.title')} ...>
      <TimelineZone events={events.data ?? { today: [], tomorrow: [], this_week: [], next_week: [] }} isLoading={events.isLoading} ... />
    </ZoneCollapsible>,
    engagements: <ZoneCollapsible title={t('zones.engagements.title')} ...>
      <EngagementsZone stages={stages.data ?? []} isLoading={stages.isLoading} ... />
    </ZoneCollapsible>,
    stats: <QuickStatsBar stats={stats.data ?? { active_engagements: 0, open_tasks: 0, sla_at_risk: 0, upcoming_week: 0 }} isLoading={stats.isLoading} />,
    activity: <ZoneCollapsible title={t('zones.activity.title')} ...>
      <ActivityFeed items={activity.data ?? []} isLoading={activity.isLoading} ... />
    </ZoneCollapsible>,
  }

  // Role-based zone ordering per D-09 and ZONE_ORDER constant
  const orderedZones = ZONE_ORDER[role]
  // On mobile, first zone in role order gets defaultExpanded: true, others false
  // On desktop, all zones render without collapsible wrapper (ZoneCollapsible passes through)

  return (
    <div dir={isRTL ? 'rtl' : 'ltr'} className="..." role="main">
      <ActionBar role={role} onRoleChange={setRole} />
      <div className="grid grid-cols-1 md:grid-cols-2 gap-6 px-4 sm:px-6 lg:px-8">
        {/* First render full-width zones, then 2-column zones */}
      </div>
    </div>
  )
}
```

**Layout logic per UI-SPEC Layout Contract:**

- Desktop (md+): grid with `grid-cols-1 md:grid-cols-2 gap-6`
- ActionBar: always `col-span-full`
- Zone rendering depends on role order, but the grid placement is:
  - The FIRST zone in role order: `col-span-full` (prominent, full width)
  - Timeline + Engagements: when both are in positions 2-3 of the role order, they sit side-by-side (each `col-span-1`). Otherwise they each get `col-span-full`
  - QuickStats: always `col-span-full` (inner grid handles 4-card layout)
  - ActivityFeed: always `col-span-full`

For the default Officer view: Attention (`col-span-full`), then Timeline (`col-span-1`) + Engagements (`col-span-1`) side by side, then Stats (`col-span-full`), then Activity (`col-span-full`). This matches the UI-SPEC desktop layout diagram.

For Leadership: Engagements (`col-span-full`), Stats (`col-span-full`), Attention (`col-span-full`), Timeline (`col-span-full`), Activity (`col-span-full`).

For Analyst: Timeline (`col-span-full`), Activity (`col-span-full`), Attention (`col-span-full`), Engagements (`col-span-full`), Stats (`col-span-full`).

**Implementation approach:** Create a `getZoneLayout(role, zoneKey, index)` helper that returns the correct `col-span` class. For Officer view specifically, when timeline is at index 1 and engagements at index 2, use `col-span-1` for both to achieve the 2-column layout.

**Mobile:** All zones stack vertically (grid-cols-1 only). Each wrapped in ZoneCollapsible. First zone in role order = `defaultExpanded: true`, rest = `defaultExpanded: false`.
