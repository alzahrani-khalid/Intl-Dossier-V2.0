```tsx
import { type ReactElement, useState } from 'react'
import { useTranslation } from 'react-i18next'
import { cva } from 'class-variance-authority'
import { ArrowUpRight } from 'lucide-react'
import { Button } from '@/components/ui/button'
import { LtrIsolate } from '@/components/ui/ltr-isolate'
import { DossierGlyph } from '@/components/signature-visuals'
import { usePersonalCommitments } from '@/hooks/usePersonalCommitments'
import type { CommitmentSeverity } from '@/hooks/usePersonalCommitments'
import { WidgetSkeleton } from './WidgetSkeleton'

const severityDot = cva('inline-block size-2 rounded-full', {
  variants: {
    severity: {
      red: 'bg-[var(--sla-bad)]',
      amber: 'bg-[var(--sla-risk)]',
      yellow: 'bg-[var(--sla-ok)]',
    },
  },
  defaultVariants: { severity: 'yellow' },
})

const SEVERITY_ORDER: Record<CommitmentSeverity, number> = { red: 0, amber: 1, yellow: 2 }
const DEFAULT_VISIBLE = 3

function groupMaxSeverity(items: { severity: CommitmentSeverity }[]): CommitmentSeverity {
  return items.reduce<CommitmentSeverity>((acc, c) => (SEVERITY_ORDER[c.severity] < SEVERITY_ORDER[acc] ? c.severity : acc), 'yellow')
}

export function OverdueCommitments(): ReactElement {
  const { t } = useTranslation('dashboard-widgets')
  const { data, isLoading, isError } = usePersonalCommitments()
  const [expanded, setExpanded] = useState<Record<string, boolean>>({})

  if (isLoading) return <WidgetSkeleton rows={4} />
  if (isError) {
    return (
      <section className="overdue rounded-lg border border-line p-4">
        <p className="text-sm text-ink-soft">{t('error.load_failed')}</p>
      </section>
    )
  }
  if (data == null || data.length === 0) {
    return (
      <section className="overdue rounded-lg border border-line p-4 text-sm text-ink-soft text-start">
        {t('overdue.empty')}
      </section>
    )
  }

  const sortedGroups = [...data].sort((a, b) =>
    SEVERITY_ORDER[groupMaxSeverity(a.commitments)] - SEVERITY_ORDER[groupMaxSeverity(b.commitments)],
  )

  return (
    <section
      role="region"
      aria-labelledby="overdue-heading"
      className="overdue rounded-lg border border-line bg-surface-raised p-4"
    >
      <h3 id="overdue-heading" className="text-sm font-medium text-ink mb-3">
        {t('overdue.title')}
      </h3>
      <ul className="space-y-3">
        {sortedGroups.map((group): ReactElement => {
          const isOpen = expanded[group.dossierId] === true
          const visible = isOpen ? group.commitments : group.commitments.slice(0, DEFAULT_VISIBLE)
          return (
            <li key={group.dossierId} className="overdue-group">
              <div className="flex items-center gap-2 mb-2">
                {group.dossierFlag != null && (
                  <DossierGlyph flag={group.dossierFlag} type="country" size={20} />
                )}
                <span className="text-xs font-medium text-ink text-start flex-1 truncate">
                  {group.dossierName}
                </span>
              </div>
              <ul className="space-y-2 ps-4">
                {visible.map((c): ReactElement => (
                  <li
                    key={c.id}
                    className="overdue-row flex items-center gap-3 min-h-11 rounded-md border border-line bg-surface p-2"
                  >
                    <span className={severityDot({ severity: c.severity })} aria-hidden="true" />
                    <span className="text-sm text-ink text-start truncate flex-1">{c.title}</span>
                    <LtrIsolate className="overdue-days font-mono text-xs text-ink-soft">
                      {c.daysOverdue}d
                    </LtrIsolate>
                    <span className="text-xs text-ink-soft font-mono" aria-label={t('overdue.owner')}>
                      {c.ownerInitials}
                    </span>
                    <ArrowUpRight className="size-4 text-ink-soft" aria-hidden="true" />
                  </li>
                ))}
              </ul>
              {group.commitments.length > DEFAULT_VISIBLE && (
                <Button
                  variant="ghost"
                  size="sm"
                  className="mt-1"
                  onClick={(): void =>
                    setExpanded((prev) => ({ ...prev, [group.dossierId]: !(prev[group.dossierId] ?? false) }))
                  }
                >
                  {isOpen ? t('overdue.collapse') : t('overdue.expand')}
                </Button>
              )}
            </li>
          )
        })}
      </ul>
    </section>
  )
}
```
