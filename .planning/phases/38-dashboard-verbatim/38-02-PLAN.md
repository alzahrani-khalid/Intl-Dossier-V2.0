```tsx
import { type ReactElement, useState } from 'react'
import { useTranslation } from 'react-i18next'
import { Button } from '@/components/ui/button'
import { Badge } from '@/components/ui/badge'
import { LtrIsolate } from '@/components/ui/ltr-isolate'
import { DossierGlyph } from '@/components/signature-visuals'
import { useAuth } from '@/hooks/useAuth'
import { useWeekAhead } from '@/hooks/useWeekAhead'
import type { TimelineGroup } from '@/domains/operations-hub/hooks/useUpcomingEvents'
import { WidgetSkeleton } from './WidgetSkeleton'

const DAY_GROUP_ORDER: TimelineGroup[] = ['today', 'tomorrow', 'this_week', 'next_week']
const MAX_VISIBLE_EVENTS = 5

export function WeekAhead(): ReactElement {
  const { t, i18n } = useTranslation('dashboard-widgets')
  const { user } = useAuth()
  const { data, isLoading, isError } = useWeekAhead(user?.id)
  const [expanded, setExpanded] = useState<Record<TimelineGroup, boolean>>({
    today: false, tomorrow: false, this_week: false, next_week: false,
  })

  if (isLoading) return <WidgetSkeleton rows={4} />
  if (isError || data == null) {
    return (
      <section className="week-ahead rounded-lg border border-line p-4">
        <p className="text-sm text-ink-soft">{t('error.load_failed')}</p>
      </section>
    )
  }

  const nonEmpty = DAY_GROUP_ORDER.filter((g) => (data[g]?.length ?? 0) > 0)
  if (nonEmpty.length === 0) {
    return (
      <section className="week-ahead rounded-lg border border-line p-4 text-sm text-ink-soft text-start">
        {t('weekAhead.empty')}
      </section>
    )
  }

  const fmtTime = (iso: string): string =>
    new Date(iso).toLocaleTimeString(i18n.language === 'ar' ? 'ar-SA' : 'en-US', {
      hour: '2-digit', minute: '2-digit',
    })

  return (
    <section
      role="region"
      aria-labelledby="week-ahead-heading"
      className="week-ahead rounded-lg border border-line bg-surface-raised p-4"
    >
      <h3 id="week-ahead-heading" className="text-sm font-medium text-ink mb-3">
        {t('weekAhead.title')}
      </h3>
      {nonEmpty.map((group): ReactElement => {
        const events = data[group] ?? []
        const isOpen = expanded[group]
        const visible = isOpen ? events : events.slice(0, MAX_VISIBLE_EVENTS)
        return (
          <div key={group} className="week-group mb-3">
            <h4 className="text-xs font-normal text-ink-soft mb-2 text-start">
              {t(`weekAhead.${group}`)}
            </h4>
            <ul className="space-y-2">
              {visible.map((e): ReactElement => (
                <li key={e.id} className="week-row flex items-center gap-3 min-h-11">
                  {e.counterpartFlag != null && (
                    <DossierGlyph flag={e.counterpartFlag} type="country" size={20} />
                  )}
                  <div className="flex-1 min-w-0">
                    <div className="text-sm text-ink text-start truncate">{e.title}</div>
                    <div className="text-xs text-ink-soft text-start truncate">
                      {e.counterpart} {e.location != null && `· ${e.location}`}
                    </div>
                  </div>
                  <LtrIsolate className="week-time text-xs text-ink-soft font-mono">
                    {fmtTime(e.start)} — {fmtTime(e.end)}
                  </LtrIsolate>
                  {e.status != null && <Badge>{t(`weekAhead.status.${e.status}`, e.status)}</Badge>}
                </li>
              ))}
            </ul>
            {events.length > MAX_VISIBLE_EVENTS && (
              <Button
                variant="ghost"
                size="sm"
                className="mt-2"
                onClick={(): void => setExpanded((prev) => ({ ...prev, [group]: !prev[group] }))}
              >
                {isOpen ? t('overdue.collapse') : t('overdue.expand')}
              </Button>
            )}
          </div>
        )
      })}
    </section>
  )
}
```

```tsx
import { describe, it, expect, vi } from 'vitest'
import { render, screen, fireEvent } from '@testing-library/react'
import { QueryClient, QueryClientProvider } from '@tanstack/react-query'
import { WeekAhead } from '../WeekAhead'

vi.mock('@/hooks/useWeekAhead', () => ({ useWeekAhead: vi.fn() }))

  DossierGlyph: (p: { flag: string }): JSX.Element => <span data-testid="glyph" data-flag={p.flag} />,
}))

import { useWeekAhead } from '@/hooks/useWeekAhead'

function makeEvents(n: number): unknown[] {
  return Array.from({ length: n }, (_, i) => ({
    id: `e${i}`, title: `Event ${i}`, start: '2026-04-24T09:00:00Z', end: '2026-04-24T10:00:00Z',
    counterpart: 'Ambassador X', counterpartFlag: 'SA', location: 'Riyadh', status: 'scheduled',
  }))
}

describe('WeekAhead', () => {
  it('renders 4 day group headers', () => {
    vi.mocked(useWeekAhead).mockReturnValue({
      data: { today: makeEvents(2), tomorrow: makeEvents(1), this_week: makeEvents(3), next_week: makeEvents(1) },
      isLoading: false, isError: false,
    } as never)
    const qc = new QueryClient()
    render(<QueryClientProvider client={qc}><WeekAhead /></QueryClientProvider>)
    expect(screen.getAllByRole('heading', { level: 4 })).toHaveLength(4)
  })

  it('shows at most 5 rows per group; expand reveals more', () => {
    vi.mocked(useWeekAhead).mockReturnValue({

    const { container } = render(<QueryClientProvider client={qc}><WeekAhead /></QueryClientProvider>)
```
