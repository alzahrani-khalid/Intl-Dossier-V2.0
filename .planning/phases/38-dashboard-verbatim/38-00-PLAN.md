**Step 1 — `frontend/src/hooks/useWeekAhead.ts`:**
```ts
import { useMemo } from 'react'
import {
  useUpcomingEvents,
  groupEventsByDay,
  type GroupedEvents,
} from '@/domains/operations-hub/hooks/useUpcomingEvents'

export function useWeekAhead(userId?: string): {
  data: GroupedEvents | undefined
  isLoading: boolean
  isError: boolean
} {
  const query = useUpcomingEvents(userId)
  const grouped = useMemo((): GroupedEvents | undefined => {
    if (query.data == null) return undefined
    return groupEventsByDay(query.data)
  }, [query.data])
  return { data: grouped, isLoading: query.isLoading, isError: query.isError }
}
```

```ts
import { useMemo } from 'react'
import { useAuth } from '@/hooks/useAuth'
import { useCommitments } from '@/hooks/useCommitments'

export type CommitmentSeverity = 'red' | 'amber' | 'yellow'

export interface GroupedCommitment {
  dossierId: string
  dossierName: string
  dossierFlag?: string
  commitments: Array<{
    id: string
    title: string
    daysOverdue: number
    severity: CommitmentSeverity
    ownerInitials: string
  }>
}

function deriveSeverity(daysOverdue: number): CommitmentSeverity {
  if (daysOverdue >= 7) return 'red'
  if (daysOverdue >= 3) return 'amber'
  return 'yellow'
}

function deriveInitials(displayName: string): string {
  return displayName
    .split(' ')
    .map((p) => p[0] ?? '')
    .join('')
    .toUpperCase()
    .slice(0, 2)
}

export function usePersonalCommitments(): {
  data: GroupedCommitment[] | undefined
  isLoading: boolean
  isError: boolean
} {
  const { user } = useAuth()
  const query = useCommitments({
    ownerId: user?.id,
    ownerType: 'internal', // CRITICAL — Pitfall 8 / T-38-09; external contacts leak otherwise
    status: 'open',
    enabled: user?.id != null,
  })
  const grouped = useMemo((): GroupedCommitment[] | undefined => {
    if (query.data == null) return undefined
    const byDossier = new Map<string, GroupedCommitment>()
    for (const c of query.data.commitments) {
      const daysOverdue = Math.max(
        0,
        Math.floor((Date.now() - new Date(c.due_date).getTime()) / 86_400_000),
      )
      if (daysOverdue === 0) continue
      const existing = byDossier.get(c.dossier_id) ?? {
        dossierId: c.dossier_id,
        dossierName: c.dossier_name ?? c.dossier_id,
        dossierFlag: c.dossier_flag,
        commitments: [],
      }
      existing.commitments.push({
        id: c.id,
        title: c.title,
        daysOverdue,
        severity: deriveSeverity(daysOverdue),
        ownerInitials: deriveInitials(c.owner_display_name ?? ''),
      })
      byDossier.set(c.dossier_id, existing)
    }
    return Array.from(byDossier.values())
  }, [query.data])
  return { data: grouped, isLoading: query.isLoading, isError: query.isError }
}
```
