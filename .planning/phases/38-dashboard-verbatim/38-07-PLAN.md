```tsx
import { type ReactElement } from 'react'
import { useTranslation } from 'react-i18next'
import { isToday, isPast, format } from 'date-fns'
import { ar, enUS } from 'date-fns/locale'
import { Checkbox } from '@/components/ui/checkbox'
import { Badge } from '@/components/ui/badge'
import { DossierGlyph } from '@/components/signature-visuals'
import { useAuth } from '@/hooks/useAuth'
import { useTasks, useUpdateTaskStatus } from '@/hooks/useTasks'
import type { WorkItem } from '@/types/work-item.types'
import { WidgetSkeleton } from './WidgetSkeleton'

function dueLabel(
  deadline: string | null,
  t: (k: string) => string,
  lang: string,
): { text: string; intent: 'default' | 'destructive' } {
  if (deadline == null) return { text: '', intent: 'default' }
  const d = new Date(deadline)
  if (isPast(d) && !isToday(d)) return { text: t('tasks.due.overdue'), intent: 'destructive' }
  if (isToday(d)) return { text: t('tasks.due.today'), intent: 'default' }
  const locale = lang === 'ar' ? ar : enUS
  return { text: format(d, 'MMM d', { locale }), intent: 'default' }
}

export function MyTasks(): ReactElement {
  const { t, i18n } = useTranslation('dashboard-widgets')
  const { user } = useAuth()
  const { data, isLoading, isError } = useTasks({ assignee_id: user?.id })
  const updateStatus = useUpdateTaskStatus()

  if (isLoading) return <WidgetSkeleton rows={4} />
  if (isError) {
    return (
      <section className="tasks rounded-lg border border-line p-4">
        <p className="text-sm text-ink-soft">{t('error.load_failed')}</p>
      </section>
    )
  }

  const tasks = data?.tasks ?? []
  if (tasks.length === 0) {
    return (
      <section className="tasks rounded-lg border border-line p-4 text-sm text-ink-soft text-start">
        {t('tasks.empty')}
      </section>
    )
  }

  const handleToggle = (task: WorkItem, checked: boolean): void => {
    void updateStatus.mutateAsync({
      id: task.id,
      status: checked ? 'completed' : 'pending',
    })
  }

  return (
    <section
      role="region"
      aria-labelledby="tasks-heading"
      className="tasks rounded-lg border border-line bg-surface-raised p-4"
    >
      <h3 id="tasks-heading" className="text-sm font-medium text-ink mb-3 text-start">
        {t('tasks.title')}
      </h3>
      <ul className="space-y-2">
        {tasks.slice(0, 6).map((task): ReactElement => {
          const done = task.status === 'completed'
          const due = dueLabel(task.deadline, t, i18n.language)
          return (
            <li
              key={task.id}
              className={`task-row flex items-center gap-3 min-h-11 ${done ? 'opacity-60 line-through' : ''}`}
            >
              <Checkbox
                checked={done}
                onCheckedChange={(c): void => handleToggle(task, c === true)}
                aria-label={task.title}
              />
              {task.dossier_flag != null && (
                <DossierGlyph flag={task.dossier_flag} type="country" size={20} />
              )}
              <span className="text-sm text-ink text-start truncate flex-1">{task.title}</span>
              {due.text !== '' && (
                <Badge variant={due.intent === 'destructive' ? 'destructive' : 'default'}>
                  {due.text}
                </Badge>
              )}
            </li>
          )
        })}
      </ul>
    </section>
  )
}
```
