import { type ReactElement } from 'react'
import { useTranslation } from 'react-i18next'
import { Link } from '@tanstack/react-router'
import { formatDistanceToNow } from 'date-fns'
import { ar, enUS } from 'date-fns/locale'
import { DossierGlyph } from '@/components/signature-visuals'
import { useDossierStore } from '@/store/dossierStore'

const MAX_RECENT = 7

export function RecentDossiers(): ReactElement {
  const { t, i18n } = useTranslation('dashboard-widgets')
  const recentDossiers = useDossierStore((s) => s.recentDossiers)
  const locale = i18n.language === 'ar' ? ar : enUS

  if (recentDossiers.length === 0) {
    return (
      <section className="recent rounded-lg border border-line p-4 text-sm text-ink-soft text-start">
        {t('recent.empty')}
      </section>
    )
  }

  return (
    <section
      role="region"
      aria-labelledby="recent-heading"
      className="recent rounded-lg border border-line bg-surface-raised p-4"
    >
      <h3 id="recent-heading" className="text-sm font-medium text-ink mb-3 text-start">
        {t('recent.title')}
      </h3>
      <ul className="space-y-2">
        {recentDossiers.slice(0, MAX_RECENT).map((d): ReactElement => (
          <li key={d.id}>
            <Link
              to={d.route}
              className="recent-row flex items-center gap-3 min-h-11 rounded-md hover:bg-surface/60 ps-2 pe-2"
            >
              {d.flag != null && <DossierGlyph flag={d.flag} type={d.type} size={20} />}
              <span className="text-sm text-ink text-start truncate flex-1">{d.name}</span>
              <span className="text-xs text-ink-soft">
                {formatDistanceToNow(d.viewedAt, { addSuffix: true, locale })}
              </span>
            </Link>
          </li>
        ))}
      </ul>
    </section>
  )
}

```tsx
import { type ReactElement } from 'react'
import { useTranslation } from 'react-i18next'
import { Badge } from '@/components/ui/badge'
import { LtrIsolate } from '@/components/ui/ltr-isolate'
import { useForums } from '@/hooks/useForums'
import { WidgetSkeleton } from './WidgetSkeleton'

function monogram(name: string): string {
  return name
    .split(/\s+/)
    .filter((w) => w.length > 0)
    .map((w) => w[0])
    .join('')
    .toUpperCase()
    .slice(0, 3)
}

export function ForumsStrip(): ReactElement {
  const { t } = useTranslation('dashboard-widgets')
  const { data, isLoading, isError } = useForums({ limit: 4 })

  if (isLoading) return <WidgetSkeleton rows={1} />
  if (isError) {
    return (
      <section className="forums rounded-lg border border-line p-4">
        <p className="text-sm text-ink-soft">{t('error.load_failed')}</p>
      </section>
    )
  }

  const forums = data?.forums ?? []
  if (forums.length === 0) {
    return (
      <section className="forums rounded-lg border border-line p-4 text-sm text-ink-soft text-start">
        {t('forums.empty')}
      </section>
    )
  }

  return (
    <section
      role="region"
      aria-labelledby="forums-heading"
      className="forums rounded-lg border border-line bg-surface-raised p-4"
    >
      <h3 id="forums-heading" className="text-sm font-medium text-ink mb-3 text-start">
        {t('forums.title')}
      </h3>
      <ul className="forums-row flex flex-col sm:flex-row gap-2">
        {forums.map((f): ReactElement => (
          <li
            key={f.id}
            className="forum-card flex items-center gap-2 rounded-md border border-line bg-surface p-2 min-h-11 flex-1 min-w-0"
          >
            <LtrIsolate className="forum-monogram font-mono text-xs text-ink">
              {monogram(f.name)}
            </LtrIsolate>
            <span className="text-sm text-ink text-start truncate flex-1">{f.name}</span>
            <Badge>{t(`forums.status.${f.status}`, f.status)}</Badge>
          </li>
        ))}
      </ul>
    </section>
  )
}
```
