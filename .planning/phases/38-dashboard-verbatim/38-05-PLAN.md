```tsx
import { type ReactElement, useMemo } from 'react'
import { useTranslation } from 'react-i18next'
import { Donut, Sparkline } from '@/components/signature-visuals'
import { useAuth } from '@/hooks/useAuth'
import { useDashboardStats } from '@/domains/operations-hub/hooks/useDashboardStats'
import { useDashboardTrends } from '@/hooks/useDashboardTrends'
import { WidgetSkeleton } from './WidgetSkeleton'

export function SlaHealth(): ReactElement {
  const { t } = useTranslation('dashboard-widgets')
  const { user } = useAuth()
  const stats = useDashboardStats(user?.id)
  const trends = useDashboardTrends('30d') // RESEARCH Pitfall 9: pass '30d' and slice 14

  const sparkSeries = useMemo((): number[] => {
    if (trends.data == null) return []
    // Concrete mapping per DASH-05: net open SLA load per day (created - completed)
    return trends.data.slice(-14).map((d) => Math.max(0, d.created - d.completed))
  }, [trends.data])

  if (stats.isLoading || trends.isLoading) return <WidgetSkeleton rows={3} />
  if (stats.isError || trends.isError || stats.data == null) {
    return (
      <section className="sla rounded-lg border border-line p-4">
        <p className="text-sm text-ink-soft">{t('error.load_failed')}</p>
      </section>
    )
  }

  const ok = stats.data.sla_ok ?? 0
  const risk = stats.data.sla_risk ?? 0
  const bad = stats.data.sla_bad ?? 0

  const segments = [
    { value: ok,   color: 'var(--sla-ok)' },
    { value: risk, color: 'var(--sla-risk)' },
    { value: bad,  color: 'var(--sla-bad)' },
  ]

  const legend = [
    { label: t('sla.legend_ok'),   count: ok,   color: 'var(--sla-ok)' },
    { label: t('sla.legend_risk'), count: risk, color: 'var(--sla-risk)' },
    { label: t('sla.legend_bad'),  count: bad,  color: 'var(--sla-bad)' },
  ]

  return (
    <section
      role="region"
      aria-labelledby="sla-heading"
      className="sla rounded-lg border border-line bg-surface-raised p-4"
    >
      <h3 id="sla-heading" className="text-sm font-medium text-ink mb-3 text-start">
        {t('sla.title')}
      </h3>
      <div className="sla-body flex items-center gap-4">
        <Donut segments={segments} size={84} stroke={10} />
        <ul className="sla-legend flex-1 space-y-1">
          {legend.map((row, i): ReactElement => (
            <li key={i} className="flex items-center gap-2 text-xs">
              <span
                className="inline-block size-2 rounded-full"
                style={{ backgroundColor: row.color }}
                aria-hidden="true"
              />
              <span className="flex-1 text-ink text-start">{row.label}</span>
              <span className="font-mono text-ink-soft">{row.count}</span>
            </li>
          ))}
        </ul>
      </div>
      <div className="sla-spark mt-3">
        {/* NO parent scaleX(-1) — Sparkline auto-flips in RTL via Phase 37 37-08-04 */}
        <Sparkline data={sparkSeries} w={80} h={22} />
      </div>
    </section>
  )
}
```
