Substitute `<CONFIRMED_HOOK>` with the hook identifier provided. Template:

```tsx
import { type ReactElement, useState } from 'react'
import { useTranslation } from 'react-i18next'
import { RefreshCcw } from 'lucide-react'
import { Button } from '@/components/ui/button'
import { Badge } from '@/components/ui/badge'
import { GlobeSpinner } from '@/components/signature-visuals'
import { <CONFIRMED_HOOK> } from '<CONFIRMED_HOOK_FILE_PATH>'
import { WidgetSkeleton } from './WidgetSkeleton'

export function Digest(): ReactElement {
  const { t, i18n } = useTranslation('dashboard-widgets')
  const { data, isLoading, isError, refetch, isFetching } = <CONFIRMED_HOOK>()
  const [clicked, setClicked] = useState(false)

  const busy = isFetching || clicked

  const handleRefresh = async (): Promise<void> => {
    setClicked(true)
    try {
      await refetch()
    } finally {
      setClicked(false)
    }
  }

  if (isLoading) return <WidgetSkeleton rows={4} />
  if (isError) {
    return (
      <section className="digest rounded-lg border border-line p-4">
        <p className="text-sm text-ink-soft">{t('error.load_failed')}</p>
      </section>
    )
  }

  const rows = data ?? []

  return (
    <section
      role="region"
      aria-labelledby="digest-heading"
      className="digest relative rounded-lg border border-line bg-surface-raised p-4"
    >
      <div className="flex items-center justify-between mb-3">
        <h3 id="digest-heading" className="text-sm font-medium text-ink text-start">
          {t('digest.title')}
        </h3>
        <Button
          variant="ghost"
          size="sm"
          onClick={(): void => { void handleRefresh() }}
          aria-label={t('digest.refresh')}
          disabled={busy}
        >
          <RefreshCcw className={`size-4 ${busy ? 'animate-spin' : ''}`} aria-hidden="true" />
        </Button>
      </div>

      {rows.length === 0 ? (
        <p className="text-sm text-ink-soft text-start">{t('digest.empty')}</p>
      ) : (
        <ul className="divide-y divide-line">
          {rows.map((r): ReactElement => (
            <li key={r.id} className="digest-row py-2 flex items-start gap-3 min-h-11">
              <Badge>{r.tag}</Badge>
              <div className="flex-1 min-w-0">
                <div className="text-sm text-ink text-start">{r.headline}</div>
                <div className="text-xs text-ink-soft text-start">
                  {r.source} · {new Date(r.timestamp).toLocaleString(i18n.language === 'ar' ? 'ar-SA' : 'en-US')}
                </div>
              </div>
            </li>
          ))}
        </ul>
      )}

      {busy && (
        <div
          aria-busy="true"
          className="digest-overlay"
          style={{
            position: 'absolute', inset: 0,
            background: 'color-mix(in srgb, var(--surface) 70%, transparent)',
            backdropFilter: 'blur(1px)',
            display: 'grid', placeItems: 'center', zIndex: 2,
          }}
        >
          <GlobeSpinner size={28} />
        </div>
      )}
    </section>
  )
}
```

**Constraints:**
- Do NOT wrap `<GlobeSpinner>` in extra `scaleX(-1)` — it is not mirror-sensitive and honors reduced-motion internally (RESEARCH Pitfall 1 + VIZ-03)
- Overlay uses `color-mix(in srgb, var(--surface) 70%, transparent)` — not a hardcoded color
