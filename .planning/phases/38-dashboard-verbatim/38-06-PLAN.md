```tsx
import { type ReactElement } from 'react'
import { useTranslation } from 'react-i18next'
import { differenceInDays } from 'date-fns'
import { Link } from '@tanstack/react-router'
import { ArrowRight } from 'lucide-react'
import { LtrIsolate } from '@/components/ui/ltr-isolate'
import { DossierGlyph } from '@/components/signature-visuals'
import { useAuth } from '@/hooks/useAuth'
import { <CONFIRMED_VIP_HOOK> } from '<CONFIRMED_VIP_HOOK_FILE_PATH>'
import { WidgetSkeleton } from './WidgetSkeleton'

export function VipVisits(): ReactElement {
  const { t, i18n } = useTranslation('dashboard-widgets')
  const { user } = useAuth()
  const { data, isLoading, isError } = <CONFIRMED_VIP_HOOK>(user?.id)
  const isRTL = i18n.language === 'ar'

  if (isLoading) return <WidgetSkeleton rows={4} />
  if (isError) {
    return (
      <section className="vip rounded-lg border border-line p-4">
        <p className="text-sm text-ink-soft">{t('error.load_failed')}</p>
      </section>
    )
  }

  // If checkpoint Choice A, filter client-side here:
  const visits = (data ?? [])
    // .filter((e) => e.event_type === '<EVENT_TYPE_VALUE>')  // only if Choice A
    .slice(0, 6)

  if (visits.length === 0) {
    return (
      <section className="vip rounded-lg border border-line p-4 text-sm text-ink-soft text-start">
        {t('vip.empty')}
      </section>
    )
  }

  const countdownLabel = (when: string): string => {
    const delta = differenceInDays(new Date(when), new Date())
    if (delta === 0) return 'T−0'
    if (delta > 0) return `T−${delta}`
    return `T+${Math.abs(delta)}`
  }

  return (
    <section
      role="region"
      aria-labelledby="vip-heading"
      className="vip rounded-lg border border-line bg-surface-raised p-4"
    >
      <div className="flex items-center justify-between mb-3">
        <h3 id="vip-heading" className="text-sm font-medium text-ink text-start">
          {t('vip.title')}
        </h3>
        <Link to="/vip-visits" className="text-xs text-ink-soft inline-flex items-center gap-1">
          {t('common.all')}
          <ArrowRight className={`size-3 ${isRTL ? 'rotate-180' : ''}`} aria-hidden="true" />
        </Link>
      </div>
      <ul className="space-y-2">
        {visits.map((v): ReactElement => (
          <li key={v.id} className="vip-row flex items-center gap-3 min-h-11">
            {v.personFlag != null && <DossierGlyph flag={v.personFlag} type="person" size={20} />}
            <div className="flex-1 min-w-0">
              <div className="text-sm text-ink text-start truncate">{v.name}</div>
              <div className="text-xs text-ink-soft text-start truncate">{v.role}</div>
            </div>
            <LtrIsolate className="vip-countdown-n text-xs font-mono text-ink-soft">
              {countdownLabel(v.when)}
            </LtrIsolate>
          </li>
        ))}
      </ul>
    </section>
  )
}
```
