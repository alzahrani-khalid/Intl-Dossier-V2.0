- `photo_url`: **File picker with thumbnail preview** (per D-03). Implement as:

  Use a hidden `<input type="file" accept="image/*">` triggered by a button. On file selection:
  1. Create a local blob URL via `URL.createObjectURL(file)` for immediate thumbnail preview
  2. Upload the file to Supabase Storage using the existing `useUploadStore.getState().uploadFile(file, { bucket: 'person-photos' })` helper
  3. On successful upload, call `form.setValue('photo_url', returnedPublicUrl)` to store the permanent URL
  4. On upload error, show the local preview still but display an error message

  ```tsx
  // Inside the component:
  const fileInputRef = useRef<HTMLInputElement>(null)
  const [previewUrl, setPreviewUrl] = useState<string | null>(null)
  const [isUploading, setIsUploading] = useState(false)
  const { uploadFile } = useUploadStore()

  const handleFileSelect = useCallback(
    async (e: React.ChangeEvent<HTMLInputElement>): Promise<void> => {
      const file = e.target.files?.[0]
      if (file == null) return

      // Show immediate preview
      const localUrl = URL.createObjectURL(file)
      setPreviewUrl(localUrl)
      setIsUploading(true)

      try {
        const publicUrl = await uploadFile(file, { bucket: 'person-photos' })
        form.setValue('photo_url', publicUrl)
      } catch {
        // Keep local preview, user can retry or remove
        console.error('Photo upload failed')
      } finally {
        setIsUploading(false)
      }
    },
    [form, uploadFile],
  )

  const handleRemovePhoto = useCallback((): void => {
    setPreviewUrl(null)
    form.setValue('photo_url', '')
    if (fileInputRef.current != null) {
      fileInputRef.current.value = ''
    }
  }, [form])
  ```

