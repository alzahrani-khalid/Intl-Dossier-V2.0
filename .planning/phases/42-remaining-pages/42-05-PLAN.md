export default function BriefsPage(): React.JSX.Element {
  const { t, i18n } = useTranslation('briefs-page')
  const locale = i18n.language as 'en' | 'ar'
  const isRTL = locale === 'ar'

  // Preserve existing dual-table fetch verbatim (lines 114-190 of original).
  // Yields: const { data: briefs, isLoading, error } = useQuery(...)
  // Each merged brief has: id, title_en, title_ar, is_published, published_date, created_at, author, full_content_en, ...

  const [selectedBrief, setSelectedBrief] = useState<MergedBrief | null>(null)
  const viewerDisclosure = useDisclosure()
  const generatorDisclosure = useDisclosure()

  function openViewer(b: MergedBrief): void {
    setSelectedBrief(b)
    viewerDisclosure.onOpen()
  }

  function chipClassFor(b: MergedBrief): string {
    if (b.is_published === true) return 'chip-ok'
    return ''
  }
  function statusKeyFor(b: MergedBrief): 'ready' | 'draft' {
    return b.is_published === true ? 'ready' : 'draft'
  }
  function pageCount(b: MergedBrief): string {
    if (!b.full_content_en) return '—'
    return String(Math.max(1, Math.ceil(b.full_content_en.length / 2200)))
  }
  function formatDayFirst(date: string | null | undefined): string {
    if (!date) return '—'
    const d = new Date(date)
    // Day-first 'Tue 28 Apr' format per CLAUDE.md
    return d.toLocaleDateString('en-GB', { weekday: 'short', day: '2-digit', month: 'short' })
  }

  return (
    <section
      role="region"
      aria-label={t('title')}
      dir={isRTL ? 'rtl' : 'ltr'}
      data-loading={isLoading ? 'true' : 'false'}
      className="page flex min-w-0 flex-col gap-[var(--gap)]"
    >
      <PageHeader
        title={t('title')}
        subtitle={t('subtitle')}
        actions={
          <button
            className="btn btn-primary"
            style={{ minHeight: 44, minWidth: 44 }}
            onClick={() => generatorDisclosure.onOpen()}
          >
            <Icon name="plus" size={14} aria-hidden />{' '}
            <span className="ms-2">{t('cta.newBrief')}</span>
          </button>
        }
      />

      {error && (
        <div className="card" role="alert">
          <Icon name="alert" size={16} style={{ color: 'var(--danger)' }} />
          <span className="ms-2">{t('error.list')}</span>
        </div>
      )}

      {isLoading && (
        <ul
          className="briefs-grid"
          style={{
            display: 'grid',
            gridTemplateColumns: 'repeat(auto-fill, minmax(320px, 1fr))',
            gap: 'var(--gap)',
            listStyle: 'none',
            padding: 0,
            margin: 0,
          }}
          data-testid="briefs-loading"
        >
          {Array.from({ length: 6 }).map((_, i) => (
            <li key={i} className="card animate-pulse" style={{ minHeight: 120 }}>
              <div className="h-3 w-1/4 bg-[var(--line-soft)] rounded" />
              <div className="mt-3 h-4 w-3/4 bg-[var(--line-soft)] rounded" />
              <div className="mt-3 h-3 w-1/2 bg-[var(--line-soft)] rounded" />
            </li>
          ))}
        </ul>
      )}

      {!isLoading && (briefs?.length ?? 0) === 0 && (
        <div className="text-center py-12 text-[var(--ink-mute)]">
          <h2 className="text-lg" style={{ fontFamily: 'var(--font-display)' }}>
            {t('empty.heading')}
          </h2>
        </div>
      )}

      {!isLoading && (briefs?.length ?? 0) > 0 && (
        <ul
          className="briefs-grid"
          style={{
            display: 'grid',
            gridTemplateColumns: 'repeat(auto-fill, minmax(320px, 1fr))',
            gap: 'var(--gap)',
            listStyle: 'none',
            padding: 0,
            margin: 0,
          }}
          data-testid="briefs-card-grid"
        >
          {briefs!.map((b) => (
            <li
              key={b.id}
              className="card"
              data-testid="brief-card"
              role="button"
              tabIndex={0}
              style={{ cursor: 'pointer', minHeight: 44 }}
              onClick={() => openViewer(b)}
              onKeyDown={(e) => {
                if (e.key === 'Enter' || e.key === ' ') openViewer(b)
              }}
            >
              <div className="flex items-center justify-between mb-2">
                <span className={cn('chip', chipClassFor(b))}>
                  {t(`status.${statusKeyFor(b)}`)}
                </span>
                <span
                  style={{
                    fontFamily: 'var(--font-mono)',
                    fontSize: 11,
                    color: 'var(--ink-faint)',
                  }}
                  dir="ltr"
                >
                  {toArDigits(`${pageCount(b)} pp`, locale)}
                </span>
              </div>
              <h3
                className="text-start"
                style={{
                  fontFamily: 'var(--font-display)',
                  fontSize: 16,
                  fontWeight: 500,
                  letterSpacing: '-0.01em',
                  lineHeight: 1.3,
                  marginBlockEnd: 10,
                }}
              >
                {locale === 'ar' ? b.title_ar || b.title_en : b.title_en || b.title_ar}
              </h3>
              <div
                className="flex items-center justify-between"
                style={{ fontSize: 11.5, color: 'var(--ink-mute)' }}
              >
                <span>{b.author?.full_name ?? '—'}</span>
                <span style={{ fontFamily: 'var(--font-mono)' }} dir="ltr">
                  {toArDigits(formatDayFirst(b.published_date ?? b.created_at), locale)}
                </span>
              </div>
            </li>
          ))}
        </ul>
      )}

      {/* BriefViewer Modal — D-05 */}
      <Modal
        isOpen={viewerDisclosure.isOpen}
        onOpenChange={viewerDisclosure.onOpenChange}
        size="3xl"
      >
        <ModalContent>
          {(onClose) => (
            <>
              <ModalHeader />
              <ModalBody>
                {selectedBrief && <BriefViewer brief={selectedBrief} onClose={onClose} />}

            </>

        </ModalContent>
      </Modal>

      {/* BriefGenerationPanel Modal — D-05 */}

        isOpen={generatorDisclosure.isOpen}

        <ModalContent>
          {(onClose) => (

              <ModalHeader />

                    onClose() /* refetch handled by query invalidation in panel */

            </>

      </Modal>
    </section>
  )
}
