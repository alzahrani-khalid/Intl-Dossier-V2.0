```tsx
import { Trans, useTranslation } from 'react-i18next'
import { useNavigate } from '@tanstack/react-router'
import { Icon, type IconName } from '@/components/signature-visuals'
import { toArDigits } from '@/lib/i18n/toArDigits'
import type { ActivityItem, ActivityActionType } from '@/types/activity-feed.types'

interface ActivityListProps {
  activities: ActivityItem[]
}

function iconForAction(t: ActivityActionType | string): IconName {
  switch (t) {
    case 'approval':
      return 'check'
    case 'rejection':
      return 'alert'
    case 'comment':
    case 'mention':
      return 'chat'
    case 'create':
      return 'plus'
    case 'delete':
      return 'alert'
    case 'upload':
    case 'download':
      return 'file'
    case 'share':
      return 'link'
    case 'update':
    case 'status_change':
    case 'view':
    case 'assign':
    case 'archive':
    case 'restore':
      return 'dot'
    default:
      return 'dot'
  }
}

function formatRelativeTime(iso: string, locale: 'en' | 'ar'): string {
  const now = Date.now()
  const then = new Date(iso).getTime()
  const diffSec = Math.max(0, Math.floor((now - then) / 1000))
  if (diffSec < 60) return locale === 'ar' ? `${diffSec}ث` : `${diffSec}s`
  const diffMin = Math.floor(diffSec / 60)
  if (diffMin < 60) return locale === 'ar' ? `${diffMin}د` : `${diffMin}m`
  const diffH = Math.floor(diffMin / 60)
  if (diffH < 24) return locale === 'ar' ? `${diffH}س` : `${diffH}h`
  const diffD = Math.floor(diffH / 24)
  return locale === 'ar' ? `${diffD}ي` : `${diffD}d`
}

export function ActivityList({ activities }: ActivityListProps): React.JSX.Element {
  const { i18n } = useTranslation('activity-feed')
  const locale = i18n.language as 'en' | 'ar'
  const navigate = useNavigate()

  return (
    <ul className="act-list">
      {activities.map((a) => {
        const navUrl = a.metadata?.navigation_url
        // R-05 open-redirect guard: only allow in-app paths (must start with '/').
        // Absolute URLs (https:, //, javascript:, etc.) and empty strings are rejected.
        const safeNavUrl = typeof navUrl === 'string' && navUrl.startsWith('/') ? navUrl : null
        const interactive = safeNavUrl !== null
        const entity =
          locale === 'ar'
            ? (a.entity_name_ar ?? a.entity_name_en ?? '—')
            : (a.entity_name_en ?? a.entity_name_ar ?? '—')
        const where =
          locale === 'ar'
            ? (a.related_entity_name_ar ?? a.related_entity_name_en ?? '—')
            : (a.related_entity_name_en ?? a.related_entity_name_ar ?? '—')
        return (
          <li
            key={a.id}
            className="act-row"
            data-testid="act-row"
            style={{ cursor: interactive ? 'pointer' : 'default', minHeight: 44 }}
            role={interactive ? 'button' : undefined}
            tabIndex={interactive ? 0 : undefined}
            onClick={
              interactive && safeNavUrl !== null
                ? () => {
                    void navigate({ href: safeNavUrl })
                  }
                : undefined
            }
            onKeyDown={
              interactive && safeNavUrl !== null
                ? (e) => {
                    if (e.key === 'Enter') void navigate({ href: safeNavUrl })
                  }
                : undefined
            }
          >
            <span className="act-t" dir="ltr">
              {toArDigits(formatRelativeTime(a.created_at, locale), locale)}
            </span>
            <Icon name={iconForAction(a.action_type)} size={16} aria-hidden />
            <span>
              <span className="act-who">{a.actor_name}</span>{' '}
              <span className="act-what">
                <Trans
                  i18nKey={`events.${a.action_type}` as 'events._default'}
                  ns="activity-feed"
                  defaults="acted on <entity>{{entity}}</entity> in <where>{{where}}</where>"
                  values={{ entity, where }}
                  components={{
                    entity: <strong style={{ color: 'var(--ink)', fontWeight: 500 }} />,
                    where: <span className="act-where" />,
                  }}
                />
              </span>
            </span>
          </li>
        )
      })}
    </ul>
  )
}
```

```tsx
import { useState } from 'react'
import { useTranslation } from 'react-i18next'
import { Tabs, TabsList, TabsTrigger } from '@/components/ui/tabs'
import { PageHeader } from '@/components/layout/PageHeader'
import { Icon } from '@/components/signature-visuals'
import { useActivityFeed } from '@/hooks/useActivityFeed'
import { ActivityList } from '@/components/activity-feed/ActivityList'

type ActivityTab = 'all' | 'following'

export default function ActivityPage(): React.JSX.Element {
  const { t, i18n } = useTranslation('activity-feed')
  const isRTL = i18n.language === 'ar'
  const [tab, setTab] = useState<ActivityTab>('all')

  const feed = useActivityFeed({ followed_only: tab === 'following' })
  const activities = feed.data?.pages.flatMap((p) => p.activities) ?? []

  return (
    <section
      role="region"
      aria-label="Activity"
      dir={isRTL ? 'rtl' : 'ltr'}
      data-loading={feed.isLoading ? 'true' : 'false'}
      className="page flex min-w-0 flex-col gap-[var(--gap)]"
    >
      <PageHeader title="Activity" />

      <Tabs value={tab} onValueChange={(v) => setTab(v as ActivityTab)}>
        <TabsList>
          <TabsTrigger value="all">{t('tabs.all')}</TabsTrigger>
          <TabsTrigger value="following">{t('tabs.following')}</TabsTrigger>
        </TabsList>
      </Tabs>

      {feed.error && (
        <div className="card" role="alert">
          <Icon name="alert" size={16} style={{ color: 'var(--danger)' }} aria-hidden />
          <span className="ms-2">{t('errorList')}</span>
        </div>
      )}

      {feed.isLoading && (
        <div className="card" data-testid="activity-skeleton">
          {Array.from({ length: 12 }).map((_, i) => (
            <div
              key={i}
              className="h-[var(--row-h)] w-full animate-pulse rounded-[var(--radius-sm)] bg-[var(--line-soft)] mb-2"
            />
          ))}
        </div>
      )}

      {!feed.isLoading && activities.length === 0 && (
        <div className="text-center py-12 text-[var(--ink-mute)]">
          <h2 className="text-lg" style={{ fontFamily: 'var(--font-display)' }}>
            {tab === 'all' ? t('empty.all') : t('empty.following')}
          </h2>
        </div>
      )}

      {!feed.isLoading && activities.length > 0 && (
        <div className="card">
          <ActivityList activities={activities} />
        </div>
      )}
    </section>
  )
}
```
