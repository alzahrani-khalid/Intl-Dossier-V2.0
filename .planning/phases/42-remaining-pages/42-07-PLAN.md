Rewrite `frontend/src/pages/MyTasks.tsx`. Preserve hook imports + `useWorkCreation` invocation; strip lucide-icon-pure UI:

```tsx
import { useState } from 'react'
import { useTranslation } from 'react-i18next'
import { useNavigate } from '@tanstack/react-router'
import cn from 'classnames'
import { Tabs, TabsList, TabsTrigger } from '@/components/ui/tabs'
import { Icon, DossierGlyph } from '@/components/signature-visuals'
import { PageHeader } from '@/components/layout/PageHeader'
import { useMyTasks, useContributedTasks, useUpdateTask } from '@/hooks/useTasks'
import { useWorkCreation } from '@/components/work-creation'
import { toArDigits } from '@/lib/i18n/toArDigits'
import type { Task } from '@/types/work-item.types'

type ViewType = 'assigned' | 'contributed'

export default function MyTasksPage(): React.JSX.Element {
  const { t, i18n } = useTranslation('tasks-page')
  const locale = i18n.language as 'en' | 'ar'
  const isRTL = locale === 'ar'
  const navigate = useNavigate()
  const { open: openWorkCreation } = useWorkCreation()

  const [view, setView] = useState<ViewType>('assigned')
  const assigned = useMyTasks()
  const contributed = useContributedTasks()
  const active = view === 'assigned' ? assigned : contributed
  const tasks = active.data?.tasks ?? []

  const updateTask = useUpdateTask()

  function priorityChipClass(priority: Task['priority'] | string): string {
    if (priority === 'urgent' || priority === 'high' || priority === 'critical')
      return 'chip chip-danger'
    if (priority === 'medium' || priority === 'normal') return 'chip chip-warn'
    return 'chip' // low / unknown
  }

  function toggleDone(task: Task): void {
    const nextStatus = task.status === 'completed' ? 'pending' : 'completed'
    updateTask.mutate({ id: task.id, updates: { status: nextStatus } })
  }

  function formatDueDate(deadline: string | null | undefined): string {
    if (!deadline) return '—'
    const d = new Date(deadline)
    return d.toLocaleDateString('en-GB', { day: '2-digit', month: 'short' })
  }
  function isToday(deadline: string | null | undefined): boolean {
    if (!deadline) return false
    const d = new Date(deadline)
    const now = new Date()
    return d.toDateString() === now.toDateString()
  }

  return (
    <section
      role="region"
      aria-label={t('title')}
      dir={isRTL ? 'rtl' : 'ltr'}
      data-loading={active.isLoading ? 'true' : 'false'}
      className="page flex min-w-0 flex-col gap-[var(--gap)]"
    >
      <PageHeader
        title={t('title')}
        subtitle={t('subtitle')}
        actions={
          <button
            className="btn btn-primary"
            style={{ minHeight: 44, minWidth: 44 }}
            onClick={() => openWorkCreation()}
          >
            <Icon name="plus" size={14} aria-hidden />{' '}
            <span className="ms-2">{t('cta.newTask')}</span>
          </button>
        }
      />

      <Tabs value={view} onValueChange={(v) => setView(v as ViewType)}>
        <TabsList>
          <TabsTrigger value="assigned">{t('tabs.assigned')}</TabsTrigger>
          <TabsTrigger value="contributed">{t('tabs.contributed')}</TabsTrigger>
        </TabsList>
      </Tabs>

      {active.error && (
        <div className="card" role="alert">
          <Icon name="alert" size={16} style={{ color: 'var(--danger)' }} aria-hidden />
          <span className="ms-2">{t('error.list')}</span>
        </div>
      )}

      {active.isLoading && (
        <div className="card" data-testid="tasks-skeleton">
          {Array.from({ length: 10 }).map((_, i) => (
            <div
              key={i}
              className="h-[var(--row-h)] w-full animate-pulse rounded-[var(--radius-sm)] bg-[var(--line-soft)] mb-2"
            />
          ))}
        </div>
      )}

      {!active.isLoading && tasks.length === 0 && (
        <div className="text-center py-12 text-[var(--ink-mute)]">
          <h2 className="text-lg" style={{ fontFamily: 'var(--font-display)' }}>
            {t('empty.heading')}
          </h2>
        </div>
      )}

      {!active.isLoading && tasks.length > 0 && (
        <div className="card">
          <ul className="tasks-list">
            {tasks.map((task) => {
              const isDone = task.status === 'completed'
              const dossier = task.dossier
              return (
                <li
                  key={task.id}
                  className={cn('task-row', isDone && 'is-done')}
                  style={isDone ? { opacity: 0.45, textDecoration: 'line-through' } : undefined}
                  role="button"
                  tabIndex={0}
                  onClick={() => void navigate({ to: '/tasks/$id', params: { id: task.id } })}
                  onKeyDown={(e) => {
                    if (e.key === 'Enter')
                      void navigate({ to: '/tasks/$id', params: { id: task.id } })
                  }}
                >
                  <button
                    role="checkbox"
                    aria-checked={isDone}
                    aria-label={
                      isDone
                        ? t('priority.done', { defaultValue: 'Mark not done' })
                        : t('priority.markDone', { defaultValue: 'Mark done' })
                    }
                    className={cn('task-box', isDone && 'done')}
                    style={{ minWidth: 44, minHeight: 44, padding: 15 }}
                    onClick={(e) => {
                      e.stopPropagation()
                      toggleDone(task)
                    }}
                    onKeyDown={(e) => e.stopPropagation()}
                  >
                    {isDone && (
                      <svg width="14" height="14" viewBox="0 0 14 14" aria-hidden>
                        <path d="M3 7l3 3 5-6" stroke="white" strokeWidth="2" fill="none" />
                      </svg>
                    )}
                  </button>
                  {dossier && (
                    <DossierGlyph
                      type={dossier.type}
                      iso={dossier.iso}
                      name={locale === 'ar' ? dossier.name_ar : dossier.name_en}
                      size={18}
                    />
                  )}
                  <div className="task-title">
                    <div>{locale === 'ar' ? (task.title_ar ?? task.title) : task.title}</div>
                    <small style={{ color: 'var(--ink-mute)' }}>
                      {`${locale === 'ar' ? (dossier?.name_ar ?? '—') : (dossier?.name_en ?? '—')} · ${task.work_item_type ?? '—'}`}
                    </small>
                  </div>
                  <span className={priorityChipClass(task.priority)}>
                    {t(`priority.${task.priority}`)}
                  </span>
                  <span
                    className={cn(
                      'task-due',
                      isToday(task.sla_deadline ?? task.deadline) && 'today',
                      (task.priority === 'high' || task.priority === 'urgent') && 'high',
                    )}
                    dir="ltr"
                  >
                    {toArDigits(formatDueDate(task.sla_deadline ?? task.deadline), locale)}
                  </span>
                </li>
              )
            })}
          </ul>
        </div>
      )}
    </section>
  )
}
```
