```sql
SELECT polname, polcmd, polpermissive,
       pg_get_expr(polqual, polrelid)      AS using_expr,
       pg_get_expr(polwithcheck, polrelid) AS check_expr
FROM pg_policy
JOIN pg_class ON pg_class.oid = polrelid
WHERE relname = 'positions';
```
