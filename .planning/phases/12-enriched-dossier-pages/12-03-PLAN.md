**backend/src/api/elected-officials.ts** — New Express router (per D-14, Pitfall 3):

CRITICAL: Join `dossiers` + `elected_officials` (NOT persons). The `elected_officials` table uses Class Table Inheritance extending `dossiers` directly.

```typescript
import { Router } from 'express'
import { authenticateToken } from '../middleware/auth.middleware'
// Follow countries.ts pattern for Supabase client usage

const router = Router()

// GET / — List with pagination and filters
// Query params: party, is_current_term (boolean), organization_id, page (default 1), limit (default 20), search
// SQL: SELECT d.id, d.name_en, d.name_ar, d.type, d.created_at, d.updated_at,
//   eo.office_title_en, eo.office_title_ar, eo.party_en, eo.party_ar,
//   eo.constituency_en, eo.constituency_ar, eo.is_current_term, eo.term_start, eo.term_end,
//   eo.organization_id, eo.committee_assignments
//   FROM dossiers d JOIN elected_officials eo ON eo.id = d.id
//   WHERE d.type = 'elected_official'
// Filter: if party, filter by party_en ILIKE or party_ar ILIKE
// Filter: if is_current_term, filter by eo.is_current_term = true/false
// Filter: if organization_id, filter by eo.organization_id = uuid
// Filter: if search, filter by d.name_en ILIKE or d.name_ar ILIKE
// Paginate: LIMIT + OFFSET, return total count in response
// Return: { data: ElectedOfficialRow[], total: number, page: number, limit: number }

// GET /:id — Single elected official with full data
// SQL: Same join, WHERE d.id = :id
// Return: { data: ElectedOfficialRow } or 404

// POST / — Create new elected official
// Body: { name_en, name_ar, description_en?, description_ar?, office_title_en, office_title_ar,
//   party_en?, party_ar?, constituency_en?, constituency_ar?, is_current_term, term_start?, term_end?,
//   organization_id?, committee_assignments? }
// Step 1: Insert into dossiers (type: 'elected_official', organization_id from auth)
// Step 2: Insert into elected_officials with same id
// Return: 201 + created entity

// PATCH /:id — Update elected official
// Body: partial fields
// Step 1: Update dossiers (name, description fields)
// Step 2: Update elected_officials (office, party, term, committee fields)
// Return: 200 + updated entity

export default router
```
