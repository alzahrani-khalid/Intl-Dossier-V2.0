```tsx
/**
 * Phase 37 — `<Donut>` — stacked dasharray donut + center percentage pill.
 *
 * Consumed by Phase 38 SlaHealth widget (DASH-05).
 * Analog: frontend/src/components/ui/pull-to-refresh-indicator.tsx lines 91-184 (dasharray math).
 * D-13 ban: plain <circle>, NOT framer-motion m.circle.
 *
 * Variant colors: [ok, risk, bad] map to var(--ok), var(--risk), var(--bad).
 * Track: var(--surface-raised). Center text: var(--ink) in var(--font-display).
 */
import type { ReactElement } from 'react'

export interface DonutProps {
  value: number
  variants: readonly [number, number, number]
  size?: number
  className?: string
}

export function Donut({
  value,
  variants,
  size = 72,
  className,
}: DonutProps): ReactElement {
  const r = 14
  const circumference = 2 * Math.PI * r

  // Clamp each variant to [0, 100] individually; let the sum be the caller's problem.
  const [okRaw, riskRaw, badRaw] = variants
  const ok = Math.max(0, Math.min(100, okRaw))
  const risk = Math.max(0, Math.min(100, riskRaw))
  const bad = Math.max(0, Math.min(100, badRaw))

  const segOk = (ok / 100) * circumference
  const segRisk = (risk / 100) * circumference
  const segBad = (bad / 100) * circumference

  const pct = Math.round(Math.max(0, Math.min(100, value)))

  return (
    <svg
      viewBox="0 0 36 36"
      width={size}
      height={size}
      aria-label={`${pct} percent`}
      role="img"
      className={className}
    >
      {/* Track */}
      <circle
        cx="18"
        cy="18"
        r={r}
        fill="none"
        stroke="var(--surface-raised)"
        strokeWidth="4"
      />
      {/* OK segment */}
      <circle
        cx="18"
        cy="18"
        r={r}
        fill="none"
        stroke="var(--ok)"
        strokeWidth="4"
        strokeDasharray={`${segOk} ${circumference}`}
        strokeDashoffset={0}
        transform="rotate(-90 18 18)"
      />
      {/* Risk segment */}
      <circle
        cx="18"
        cy="18"
        r={r}
        fill="none"
        stroke="var(--risk)"
        strokeWidth="4"
        strokeDasharray={`${segRisk} ${circumference}`}
        strokeDashoffset={-segOk}
        transform="rotate(-90 18 18)"
      />
      {/* Bad segment */}
      <circle
        cx="18"
        cy="18"
        r={r}
        fill="none"
        stroke="var(--bad)"
        strokeWidth="4"
        strokeDasharray={`${segBad} ${circumference}`}
        strokeDashoffset={-(segOk + segRisk)}
        transform="rotate(-90 18 18)"
      />
      {/* Center percentage pill */}
      <text
        x="18"
        y="19"
        textAnchor="middle"
        dominantBaseline="middle"
        fill="var(--ink)"
        fontSize="8"
        fontWeight="600"
        style={{ fontFamily: 'var(--font-display)' }}
      >
        {pct}%
      </text>
    </svg>
  )
}
```
