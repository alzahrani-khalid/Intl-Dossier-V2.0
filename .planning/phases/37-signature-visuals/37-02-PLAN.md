```ts
/**
 * Phase 37 — Lazy loader for d3-geo + topojson-client + world-atlas.
 * Called on first `<GlobeLoader>` mount; memoized so repeat mounts reuse the resolved promise.
 *
 * Handoff source: /tmp/inteldossier-handoff/inteldossier/project/src/loader.jsx lines 10-33
 * (handoff used UMD fetch; we use Vite dynamic import + `.default` extraction per RESEARCH Pitfall 1).
 */
import type { GeoPermissibleObjects } from 'd3-geo'

export interface EnsureWorldResult {
  countries: GeoPermissibleObjects
  graticule: GeoPermissibleObjects
  d3: typeof import('d3-geo')
}

let _worldPromise: Promise<EnsureWorldResult> | null = null

export function ensureWorld(): Promise<EnsureWorldResult> {
  if (_worldPromise !== null) return _worldPromise
  _worldPromise = (async (): Promise<EnsureWorldResult> => {
    const [d3, topojson, worldModule] = await Promise.all([
      import('d3-geo'),
      import('topojson-client'),
      import('world-atlas/countries-110m.json'),
    ])
    const world = (worldModule as { default: unknown }).default
    const countries = topojson.feature(
      world as Parameters<typeof topojson.feature>[0],
      (world as { objects: { countries: unknown } }).objects.countries as Parameters<typeof topojson.feature>[1],
    ) as unknown as GeoPermissibleObjects
    const graticule = d3.geoGraticule10() as unknown as GeoPermissibleObjects
    return { countries, graticule, d3 }
  })()
  return _worldPromise
}

// Test-only escape hatch — Plan 07 may need to reset memoization.
export function __resetEnsureWorldForTests(): void {
  _worldPromise = null
}
```

```tsx
/**
 * Phase 37 — `<GlobeLoader>`.
 * Verbatim TS port of /tmp/inteldossier-handoff/inteldossier/project/src/loader.jsx.
 *
 * Rings + halo paint from CSS immediately (progressive render per D-07).
 * d3-geo + topojson-client + world-atlas load lazily on first mount (D-05).
 * Graceful degrade: on chunk failure, rings keep spinning and a single console.warn
 * is emitted (D-08). Reduced-motion: full stop — rAF cancelled, λ pinned at 0 (D-14).
 */
import { useEffect, useRef, type ReactElement } from 'react'
import { ensureWorld } from './ensureWorld'
import { useReducedMotion } from '@/design-system/hooks'
import './globe-loader.css'

export interface GlobeLoaderProps {
  size?: number
  speed?: number   // deg/sec; default 16 (VIZ-01)
  tilt?: number    // deg; default -18 (VIZ-01)
  className?: string
}

export function GlobeLoader({
  size = 120,
  speed = 16,
  tilt = -18,
  className,
}: GlobeLoaderProps): ReactElement {
  const hostRef = useRef<HTMLDivElement>(null)
  const reducedMotion = useReducedMotion()

  useEffect((): (() => void) => {
    let raf = 0
    let cancelled = false
    ;(async (): Promise<void> => {
      const { countries, graticule, d3 } = await ensureWorld()
      if (cancelled || hostRef.current === null) return
      // Select the pre-rendered .gl-svg container and append sphere / graticule / land paths.
      // See handoff loader.jsx lines 38-80 for the exact d3 select/append sequence.
      const svg = d3.select(hostRef.current).select<SVGSVGElement>('.gl-svg')
      const projection = d3.geoOrthographic().scale(size * 0.4).translate([0, 0]).clipAngle(90)
      const path = d3.geoPath(projection as unknown as Parameters<typeof d3.geoPath>[0])
      const sphere = svg.append('path').attr('d', path({ type: 'Sphere' } as never) ?? '').attr('fill', 'none').attr('stroke', 'var(--ink)').attr('stroke-width', '0.6').attr('opacity', '0.45')
      const grat = svg.append('path').datum(graticule as never).attr('d', path as never).attr('fill', 'none').attr('stroke', 'var(--ink)').attr('stroke-width', '0.3').attr('opacity', '0.25')
      const land = svg.append('path').datum(countries as never).attr('d', path as never).attr('fill', 'var(--ink)').attr('opacity', '0.75').attr('data-globe-land', 'true')

      // Render once at λ=0 for reduced-motion users (Pitfall 4) then return.
      if (reducedMotion) {
        projection.rotate([0, tilt])
        sphere.attr('d', path({ type: 'Sphere' } as never) ?? '')
        grat.attr('d', path as never)
        land.attr('d', path as never)
        return
      }

      const start = performance.now()
      const tick = (now: number): void => {
        if (cancelled) return
        const lambda = ((now - start) / 1000) * speed % 360 - 180
        projection.rotate([lambda, tilt])
        sphere.attr('d', path({ type: 'Sphere' } as never) ?? '')
        grat.attr('d', path as never)
        land.attr('d', path as never)
        raf = requestAnimationFrame(tick)
      }
      raf = requestAnimationFrame(tick)
    })().catch((err: unknown): void => {
      console.warn('GlobeLoader: topojson failed to load', err)
    })

    return (): void => {
      cancelled = true
      if (raf !== 0) cancelAnimationFrame(raf)
    }
  }, [size, speed, tilt, reducedMotion])

  return (
```
