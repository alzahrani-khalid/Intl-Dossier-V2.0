    **Composition contract (UI-SPEC §"Layout Tokens — Grid"):**

    ```tsx
    import { useState, useEffect, type ReactNode } from 'react'
    import { useRouterState } from '@tanstack/react-router'
    import { Drawer } from '@heroui/react'
    import { cn } from '@/lib/utils'
    import { SidebarProvider } from '@/components/ui/sidebar'
    import { Sidebar } from './Sidebar'
    import { Topbar } from './Topbar'
    import { ClassificationBar } from './ClassificationBar'
    import { useDirection } from '@/hooks/useDirection' // or from '@/providers/DesignProvider'
    import { useTranslation } from 'react-i18next'

    export interface AppShellProps {
      children: ReactNode
    }

    function getStoredSidebarOpen(): boolean {
      try {
        return localStorage.getItem('sidebar_state') !== 'false'
      } catch {
        return true
      }
    }

    export function AppShell({ children }: AppShellProps): JSX.Element {
      const [drawerOpen, setDrawerOpen] = useState(false)
      const { direction } = useDirection()
      const { t } = useTranslation()
      const pathname = useRouterState({ select: (s) => s.location.pathname })

      // Auto-close drawer on route change (UI-SPEC §"Close triggers" #5)
      useEffect(() => {
        setDrawerOpen(false)
      }, [pathname])

      return (
        <SidebarProvider
          dir={direction}
          defaultOpen={getStoredSidebarOpen()}
          style={{
            '--sidebar-width': '16rem',  // 256px
            '--header-height': '3.5rem', // 56px
          } as React.CSSProperties}
        >
          <div className="appshell relative grid min-h-screen w-full grid-rows-[auto_1fr] lg:grid-cols-[16rem_1fr]">
            {/* Sidebar — desktop column 1, rows 1-3. Hidden below lg; drawer takes over */}
            <aside className="appshell-aside hidden lg:block lg:row-span-full lg:col-start-1 border-e border-[var(--line)]">
              <Sidebar />
            </aside>

            {/* Topbar — row 1, col 2 on desktop; full width on mobile */}
            <div className="appshell-top lg:col-start-2 lg:row-start-1">
              <Topbar onOpenDrawer={() => setDrawerOpen(true)} />
            </div>

            {/* Classification chrome — row 2 on desktop; between topbar and main */}
            <div className="appshell-classif relative lg:col-start-2 lg:row-start-2">
              <ClassificationBar />
            </div>

            {/* Main content — row 3 desktop; fills */}
            <main className="appshell-main lg:col-start-2 lg:row-start-3 overflow-y-auto bg-[var(--bg)]">
              {children}
            </main>
          </div>

          {/* Mobile overlay drawer — only renders/engages below lg */}
          <Drawer
            isOpen={drawerOpen}
            onOpenChange={setDrawerOpen}
            placement="left"
            size="sm"
            classNames={{
              base: 'lg:hidden',
              wrapper: 'lg:hidden',
              body: 'p-0',
            }}
          >
            <div
              className="appshell-drawer-panel w-[280px] max-sm:w-screen h-full"
              aria-label={t('shell.menu.open')}
              onClick={(e) => {
                // Auto-close on nav-item click (per UI-SPEC §"Close triggers" #4)
                const target = e.target as HTMLElement
                if (target.closest('a[href]')) setDrawerOpen(false)
              }}
            >
              <Sidebar />
            </div>
          </Drawer>
        </SidebarProvider>
      )
    }
    ```

    **Key rules (from Pitfall 2 — FOUC/flash):**
    - `AppShell` NEVER conditionally unmounts children based on direction. The inner tree stays mounted across `useDirection()` changes — `applyTokens()` paints CSS vars on `documentElement`, Tailwind utilities repaint without React commits.
    - Do NOT gate `<main>` / `<Topbar>` / `<Sidebar>` rendering on `direction` or `mode`.

    **Key rules (from Pitfall 5 — Classification chip):**
    - `<div className="appshell-classif relative ...">` gives the chip variant a positioned parent. ClassificationBar's chip uses `absolute` + `insetInlineStart: 0` to appear at page-head start.

    **Drawer close triggers (all must work):**
    1. ESC → HeroUI Drawer built-in
    2. Backdrop click → HeroUI Drawer built-in
    3. Re-click hamburger → Topbar's button is not toggle — it only calls `onOpenDrawer`. Add `toggle` semantics: if open, close; if closed, open. Change the prop name to `onToggleDrawer` OR keep `onOpenDrawer` and handle re-click idempotently.
    4. Nav-item click → handled above via event delegation on the drawer panel
    5. Route change → `useEffect(() => setDrawerOpen(false), [pathname])`

    **RTL flip:**
    - HeroUI Drawer with `placement="left"` auto-flips to right under `html[dir="rtl"]` (Phase 34 D-03 proven)
    - Classification chip uses `ms-5` / logical — flips automatically
    - `border-e` on sidebar (not `border-r`) — flips automatically

    **a11y:**
    - Drawer panel has `aria-label={t('shell.menu.open')}`
    - Topbar hamburger and drawer close share focus restoration via HeroUI Drawer built-in

    **No physical properties:**

    - Use `start-*/end-*/ms-*/me-*/ps-*/pe-*/border-s/border-e` exclusively
