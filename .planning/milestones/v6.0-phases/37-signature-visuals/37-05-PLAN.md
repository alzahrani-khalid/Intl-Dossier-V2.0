```tsx
/**
 * Phase 37 — `<DossierGlyph>` — dossier visual identity primitive.
 *
 * Resolution order (per D-09..D-11 + RESEARCH Pitfall 6):
 *   1. type='country' + iso in 24-set → hand-drawn flag clipped to circle with 1px hairline
 *   2. type='country' + unknown iso → initials fallback (1-2 chars from name)
 *   3. type in {forum, person, topic, organization} → Unicode symbol in soft-tinted circle
 *   4. type in {engagement, working_group, elected_official} → initials fallback (Pitfall 6)
 *
 * Security (T-37-01): All flag SVGs are static inline TSX authored in-repo; never fetched,
 * never string-injected. `name` prop is rendered via React text node (auto-escaped).
 */
import type { ReactElement } from 'react'
import type { DossierType } from '@/types/dossier-context.types'
import { flags, type FlagKey } from './flags'

const FLAG_KEYS: ReadonlySet<FlagKey> = new Set<FlagKey>([
  'sa',
  'ae',
  'id',
  'eg',
  'qa',
  'jo',
  'bh',
  'om',
  'kw',
  'pk',
  'ma',
  'tr',
  'cn',
  'it',
  'fr',
  'de',
  'gb',
  'us',
  'jp',
  'kr',
  'in',
  'br',
  'eu',
  'un',
])

const SYMBOL_MAP: Partial<Record<DossierType, string>> = {
  forum: '\u25C7',
  person: '\u25CF',
  topic: '\u25C6',
  organization: '\u25B2',
}

function toInitials(nameOrIso: string): string {
  const trimmed = nameOrIso.trim()
  if (trimmed.length === 0) return '?'
  const parts = trimmed.split(/\s+/)
  if (parts.length >= 2) {
    return `${(parts[0] ?? '').charAt(0)}${(parts[1] ?? '').charAt(0)}`.toUpperCase()
  }
  return trimmed.slice(0, 2).toUpperCase()
}

export interface DossierGlyphProps {
  type: DossierType
  iso?: string
  name?: string
  size?: number
  accent?: string
  className?: string
}

interface InitialsBadgeProps {
  text: string
  size: number
  accent: string
  className?: string
}

function InitialsBadge({ text, size, accent, className }: InitialsBadgeProps): ReactElement {
  return (
    <span
      className={className}
      style={{
        width: size,
        height: size,
        display: 'inline-flex',
        alignItems: 'center',
        justifyContent: 'center',
        borderRadius: '50%',
        background: `color-mix(in srgb, ${accent} 12%, var(--surface-raised))`,
        color: accent,
        fontSize: size * 0.42,
        fontWeight: 600,
      }}
      aria-hidden="true"
    >
      {text}
    </span>
  )
}

export function DossierGlyph({
  type,
  iso,
  name,
  size = 32,
  accent = 'var(--ink)',
  className,
}: DossierGlyphProps): ReactElement {
  if (type === 'country' && iso !== undefined) {
    const key = iso.toLowerCase() as FlagKey
    if (FLAG_KEYS.has(key)) {
      const FlagSvg = flags[key]
      const clipId = `dg-clip-${key}`
      return (
        <span
          className={className}
          style={{ width: size, height: size, display: 'inline-block' }}
          aria-hidden="true"
        >
          <svg viewBox="0 0 32 32" width={size} height={size}>
            <defs>
              <clipPath id={clipId}>
                <circle cx="16" cy="16" r="16" />
              </clipPath>
            </defs>
            <g clipPath={`url(#${clipId})`}>
              <FlagSvg />
            </g>
            <circle
              cx="16"
              cy="16"
              r="15.5"
              fill="none"
              stroke="rgba(0,0,0,0.15)"
              strokeWidth="1"
            />
          </svg>
        </span>
      )
    }
    return (
      <InitialsBadge
        text={toInitials(name ?? iso)}
        size={size}
        accent={accent}
        className={className}
      />
    )
  }

  const symbol = SYMBOL_MAP[type]
  if (symbol !== undefined) {
    return (
      <span
        className={className}
        style={{
          width: size,
          height: size,
          display: 'inline-flex',
          alignItems: 'center',
          justifyContent: 'center',
          borderRadius: '50%',
          background: `color-mix(in srgb, ${accent} 12%, var(--surface-raised))`,
          color: accent,
          fontSize: size * 0.5,
        }}
        aria-hidden="true"
      >
        {symbol}
      </span>
    )
  }

  return (
    <InitialsBadge
      text={toInitials(name ?? '?')}
      size={size}
      accent={accent}
      className={className}
    />
  )
}
```
