```ts
import { test as setup, expect } from '@playwright/test'
import path from 'path'

const roles = [
  { name: 'admin', email: process.env.E2E_ADMIN_EMAIL!, password: process.env.E2E_ADMIN_PASSWORD! },
  {
    name: 'analyst',
    email: process.env.E2E_ANALYST_EMAIL!,
    password: process.env.E2E_ANALYST_PASSWORD!,
  },
  {
    name: 'intake',
    email: process.env.E2E_INTAKE_EMAIL!,
    password: process.env.E2E_INTAKE_PASSWORD!,
  },
]

for (const role of roles) {
  setup(`authenticate as ${role.name}`, async ({ page }) => {
    await page.goto('/login')
    await page.getByLabel(/email/i).fill(role.email)
    await page.getByLabel(/password/i).fill(role.password)
    await page.getByRole('button', { name: /sign in|login/i }).click()
    await expect(page).toHaveURL(/dashboard|operations|home/, { timeout: 15_000 })
    await page
      .context()
      .storageState({ path: path.join('tests/e2e/support/storage', `${role.name}.json`) })
  })
}
```
