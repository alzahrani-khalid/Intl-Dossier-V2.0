       -- Notify engagement owner when lifecycle stage changes
       -- Uses existing create_categorized_notification RPC
       CREATE OR REPLACE FUNCTION notify_lifecycle_stage_change()
       RETURNS TRIGGER AS $$
       DECLARE
         v_owner_id UUID;
         v_dossier_title TEXT;
       BEGIN
         IF OLD.lifecycle_stage IS DISTINCT FROM NEW.lifecycle_stage THEN
           -- Get the dossier owner and title
           SELECT created_by, title INTO v_owner_id, v_dossier_title
           FROM dossiers WHERE id = NEW.id;

           IF v_owner_id IS NOT NULL THEN
             PERFORM create_categorized_notification(
               v_owner_id,
               'lifecycle_transition',
               'Engagement stage changed',
               'Engagement "' || COALESCE(v_dossier_title, 'Unknown') || '" moved to ' || NEW.lifecycle_stage,

               '/engagements/' || NEW.id::text,
               'engagement',
               NEW.id::text,
               NULL
             );
           END IF;
         END IF;
         RETURN NEW;
       END;
       $$ LANGUAGE plpgsql SECURITY DEFINER;

         FOR EACH ROW

| ------------------------------ | ------------------------------------------------------------------ |
| Task service -> Queue          | Backend service enqueues after validated DB operation              |
| Postgres trigger -> RPC        | SECURITY DEFINER function inserts notification on behalf of system |
| Deadline scheduler -> Supabase | Service role queries tasks table to find approaching deadlines     |

## STRIDE Threat Register

| Threat ID | Category               | Component                              | Disposition | Mitigation Plan                                                                                                                                                       |
| --------- | ---------------------- | -------------------------------------- | ----------- | --------------------------------------------------------------------------------------------------------------------------------------------------------------------- |
| T-15-05   | Spoofing               | tasks.service.ts notification dispatch | mitigate    | Notification only fires server-side after authenticated task mutation; userId comes from DB record, not user input                                                    |
| T-15-06   | Info Disclosure        | deadline-scheduler.ts                  | mitigate    | Deadline check uses service role but only reads task assignee_id and title; no sensitive data in notification payload                                                 |
| T-15-07   | Denial of Service      | deadline-scheduler.ts                  | mitigate    | jobId dedup prevents duplicate notifications; repeatable job has fixed interval (15 min); BullMQ concurrency: 5 caps parallel processing                              |
| T-15-08   | Elevation of Privilege | lifecycle trigger SQL                  | mitigate    | Function is SECURITY DEFINER but only calls create_categorized_notification for the dossier owner; no user-controlled input in the trigger                            |
| T-15-09   | Tampering              | lifecycle trigger SQL                  | accept      | Trigger fires on any lifecycle_stage UPDATE; stage values are constrained by application logic in the Edge Function; no additional validation needed at trigger level |
